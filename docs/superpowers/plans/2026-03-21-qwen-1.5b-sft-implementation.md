```python
    batch_size: int = 1

    warmup_ratio: float = 0.1

        lora_raw = model_raw.get("lora", {})
        lora_config = LoRAConfig(
            r=lora_raw.get("r", 8),
            alpha=lora_raw.get("alpha", 16),

        # 构建各子配置
        model_config = ModelConfig(
            name=model_raw.get("name", "Qwen2.5-1.5B-Instruct"),
            path=model_raw.get("path", ""),
            use_lora=model_raw.get("use_lora", True),
            lora=lora_config
        )

        training_config = TrainingConfig(
            learning_rate=training_raw.get("learning_rate", 5e-5),
            batch_size=training_raw.get("batch_size", 1),
            gradient_accumulation_steps=training_raw.get("gradient_accumulation_steps", 128),
            num_epochs=training_raw.get("num_epochs", 3),
            warmup_ratio=training_raw.get("warmup_ratio", 0.1),
            weight_decay=training_raw.get("weight_decay", 0.01),
            max_grad_norm=training_raw.get("max_grad_norm", 1.0),
            lr_scheduler_type=training_raw.get("lr_scheduler_type", "cosine"),
            logging_steps=training_raw.get("logging_steps", 10),
            save_steps=training_raw.get("save_steps", 500),
            eval_steps=training_raw.get("eval_steps", 500)
        )

        optimization_config = OptimizationConfig(
            max_length=opt_raw.get("max_length", 1024),
            gradient_checkpointing=opt_raw.get("gradient_checkpointing", True),
            mixed_precision=opt_raw.get("mixed_precision", "fp16")
        )

        data_config = DataConfig(
            base_path=data_raw.get("base_path", ""),
            train_file=data_raw.get("train_file", "splits/train.jsonl"),
            dev_file=data_raw.get("dev_file", "splits/dev.jsonl"),
            system_prompt=data_raw.get("system_prompt", "你是一个地理空间推理专家。")

        output_config = OutputConfig(
            base_dir=output_raw.get("base_dir", "outputs"),
            logging_dir=output_raw.get("logging_dir", "logs"),
            checkpoint_dir=output_raw.get("checkpoint_dir", "checkpoints")
        )

        return cls(
            experiment_name=exp.get("name", "qwen-1.5b-sft"),
            experiment_type=exp.get("type", "exp1_direct_sft"),
```

```python
# exp/exp0/qwen-1.5B-sft/src/data_processor.py
"""
数据处理模块 - 转换 GeoKD-SR 数据为 ChatML 格式
"""

import json
from pathlib import Path
from typing import Dict, List, Optional, Any
from dataclasses import dataclass
import logging

logger = logging.getLogger(__name__)


@dataclass
class ConversationSample:
    """对话样本"""
    messages: List[Dict[str, str]]
    raw_data: Dict[str, Any]  # 保留原始数据用于评测


class DataProcessor:
    """数据处理类 - 转换为 ChatML 格式"""

    def __init__(
        self,
        system_prompt: str = "你是一个地理空间推理专家，能够准确分析地理实体之间的空间关系。",
        max_length: int = 1024
    ):
        """
        初始化数据处理器

        Args:
            system_prompt: 系统提示词
            max_length: 最大序列长度
        """
        self.system_prompt = system_prompt
        self.max_length = max_length

    def load_jsonl(self, file_path: str) -> List[Dict]:
        """加载 JSONL 文件"""
        data = []
        path = Path(file_path)

        if not path.exists():
            raise FileNotFoundError(f"文件不存在: {file_path}")

        with open(path, "r", encoding="utf-8") as f:
            for line_num, line in enumerate(f, 1):
                line = line.strip()
                if not line:
                    continue
                try:
                    item = json.loads(line)
                    data.append(item)
                except json.JSONDecodeError as e:
                    logger.warning(f"第 {line_num} 行解析失败: {e}")

        logger.info(f"从 {file_path} 加载了 {len(data)} 条数据")
        return data

    def convert_to_chatml(self, item: Dict) -> ConversationSample:
        """
        将原始数据转换为 ChatML 格式

        Args:
            item: 原始数据项

        Returns:
            ConversationSample 对象
        """
        question = item.get("question", "")
        answer = item.get("answer", "")

        messages = [
            {"role": "system", "content": self.system_prompt},
            {"role": "user", "content": question},
            {"role": "assistant", "content": answer}
        ]

        return ConversationSample(messages=messages, raw_data=item)

    def process_dataset(self, data: List[Dict]) -> List[ConversationSample]:
        """处理整个数据集"""
        samples = []
        for item in data:
            try:
                sample = self.convert_to_chatml(item)
                samples.append(sample)
            except Exception as e:
                logger.warning(f"处理数据失败: {e}, item_id={item.get('id', 'unknown')}")

        logger.info(f"成功处理 {len(samples)} 条数据")
        return samples

    def prepare_for_training(
        self,
        samples: List[ConversationSample],
        tokenizer
    ) -> "Dataset":
        """
        准备训练数据集

        Args:
            samples: 对话样本列表
            tokenizer: 分词器

        Returns:
            HuggingFace Dataset
        """
        from datasets import Dataset

        def tokenize_function(sample: ConversationSample):
            """使用 ChatML 模板进行分词"""
            # 使用 tokenizer 的 apply_chat_template
            text = tokenizer.apply_chat_template(
                sample.messages,
                tokenize=False,
                add_generation_prompt=False
            )

            # 分词
            tokenized = tokenizer(
                text,
                max_length=self.max_length,
                truncation=True,
                padding=False,
                return_tensors=None
            )

            # 构造 labels：仅 assistant 部分计算损失
            # 找到 assistant 回答的位置
            assistant_text = sample.messages[-1]["content"]

            # 重新分词以获取 assistant 部分
            full_text = text
            assistant_start = full_text.rfind(assistant_text)

            if assistant_start > 0:
                # 计算 assistant 部分的 token 范围
                prefix_text = full_text[:assistant_start]
                prefix_tokens = tokenizer(prefix_text, add_special_tokens=False)["input_ids"]
                assistant_tokens = tokenizer(assistant_text, add_special_tokens=False)["input_ids"]

                # 创建 labels：prefix 部分为 -100
                labels = [-100] * len(prefix_tokens) + assistant_tokens.copy()

                # 对齐长度
                input_ids = tokenized["input_ids"]
                if len(labels) < len(input_ids):
                    labels = labels + [-100] * (len(input_ids) - len(labels))
                elif len(labels) > len(input_ids):
                    labels = labels[:len(input_ids)]

                tokenized["labels"] = labels
            else:
                # 如果找不到 assistant 位置，使用简单的 label 策略
                tokenized["labels"] = tokenized["input_ids"].copy()

            return tokenized

        # 转换为 Dataset
        processed_data = []
        for sample in samples:
            tokenized = tokenize_function(sample)
            processed_data.append({
                "input_ids": tokenized["input_ids"],
                "attention_mask": tokenized["attention_mask"],
                "labels": tokenized["labels"]
            })

        dataset = Dataset.from_list(processed_data)
        logger.info(f"创建了包含 {len(dataset)} 条样本的数据集")

        return dataset


def create_data_collator(tokenizer):
    """创建数据整理器"""
    from transformers import DataCollatorForSeq2Seq

    return DataCollatorForSeq2Seq(
```

```python
# exp/exp0/qwen-1.5B-sft/src/trainer.py
"""
训练器封装模块
"""

import torch
from pathlib import Path
from typing import Optional, Dict, Any
from dataclasses import dataclass
import logging
import sys

logger = logging.getLogger(__name__)


@dataclass
class TrainingResult:
    """训练结果"""
    checkpoint_path: str
    final_loss: float
    training_time: float
    metrics: Dict[str, Any]


class SFTTrainerWrapper:
    """SFT 训练器封装"""

    def __init__(self, config):
        """
        初始化训练器

        Args:
            config: Config 对象
        """
        self.config = config
        self.model = None
        self.tokenizer = None
        self.trainer = None

    def setup_model_and_tokenizer(self):
        """设置模型和分词器"""
        from transformers import AutoModelForCausalLM, AutoTokenizer

        logger.info(f"加载模型: {self.config.model.path}")

        # 设置 dtype
        if self.config.optimization.mixed_precision == "bf16":
            torch_dtype = torch.bfloat16
        elif self.config.optimization.mixed_precision == "fp16":
            torch_dtype = torch.float16
        else:
            torch_dtype = torch.float32

        # 加载 tokenizer
        self.tokenizer = AutoTokenizer.from_pretrained(
            self.config.model.path,
            trust_remote_code=True,
            use_fast=False
        )

        # 确保 pad_token 存在
        if self.tokenizer.pad_token is None:
            self.tokenizer.pad_token = self.tokenizer.eos_token
            self.tokenizer.pad_token_id = self.tokenizer.eos_token_id

        # 加载模型
        self.model = AutoModelForCausalLM.from_pretrained(
            self.config.model.path,
            torch_dtype=torch_dtype,
            device_map="auto",
            trust_remote_code=True,
            low_cpu_mem_usage=True
        )

        # 启用 gradient checkpointing
        if self.config.optimization.gradient_checkpointing:
            self.model.gradient_checkpointing_enable()
            logger.info("已启用 gradient checkpointing")

        # 应用 LoRA
        if self.config.model.use_lora:
            self._apply_lora()

        logger.info(f"模型加载完成，参数量: {sum(p.numel() for p in self.model.parameters()) / 1e9:.2f}B")

        return self.model, self.tokenizer

    def _apply_lora(self):
        """应用 LoRA 适配器"""
        from peft import LoraConfig, get_peft_model, TaskType

        lora_config = LoraConfig(
            task_type=TaskType.CAUSAL_LM,
            r=self.config.model.lora.r,
            lora_alpha=self.config.model.lora.alpha,
            lora_dropout=self.config.model.lora.dropout,
            target_modules=self.config.model.lora.target_modules,
            bias="none"
        )

        self.model = get_peft_model(self.model, lora_config)
        self.model.print_trainable_parameters()

        logger.info(
            f"LoRA 配置: r={lora_config.r}, "
            f"alpha={lora_config.lora_alpha}, "
            f"dropout={lora_config.lora_dropout}"
        )

    def setup_trainer(self, train_dataset, eval_dataset=None):
        """设置 Trainer"""
        from transformers import TrainingArguments, Trainer
        from .data_processor import create_data_collator

        # 创建输出目录
        output_dir = Path(self.config.output.checkpoint_dir)
        output_dir.mkdir(parents=True, exist_ok=True)

        # 训练参数
        training_args = TrainingArguments(
            output_dir=str(output_dir),
            overwrite_output_dir=True,

            # 训练超参数
            num_train_epochs=self.config.training.num_epochs,
            per_device_train_batch_size=self.config.training.batch_size,
            per_device_eval_batch_size=self.config.training.batch_size,
            gradient_accumulation_steps=self.config.training.gradient_accumulation_steps,

            # 学习率
            learning_rate=self.config.training.learning_rate,
            weight_decay=self.config.training.weight_decay,
            warmup_ratio=self.config.training.warmup_ratio,
            lr_scheduler_type=self.config.training.lr_scheduler_type,
            max_grad_norm=self.config.training.max_grad_norm,

            # 混合精度
            fp16=self.config.optimization.mixed_precision == "fp16",
            bf16=self.config.optimization.mixed_precision == "bf16",

            # 日志
            logging_dir=self.config.output.logging_dir,
            logging_steps=self.config.training.logging_steps,

            # 保存
            save_steps=self.config.training.save_steps,
            save_total_limit=3,
            load_best_model_at_end=eval_dataset is not None,

            # 评估
            evaluation_strategy="steps" if eval_dataset else "no",
            eval_steps=self.config.training.eval_steps if eval_dataset else None,

            # 其他
            seed=self.config.seed,
            dataloader_num_workers=0,  # Windows 兼容
            remove_unused_columns=False,
            label_names=["labels"],
        )

        # 数据整理器
        data_collator = create_data_collator(self.tokenizer)

        # 创建 Trainer
        self.trainer = Trainer(
            model=self.model,
            args=training_args,
            train_dataset=train_dataset,
            eval_dataset=eval_dataset,
            tokenizer=self.tokenizer,
            data_collator=data_collator,
        )

        logger.info(f"Trainer 设置完成: {training_args}")

        return self.trainer

    def train(self) -> TrainingResult:
        """执行训练"""
        import time

        logger.info("开始训练...")
        start_time = time.time()

        # 训练
        train_result = self.trainer.train()

        training_time = time.time() - start_time

        # 保存最终模型
        final_output_dir = Path(self.config.output.checkpoint_dir) / "final"
        self.trainer.save_model(str(final_output_dir))
        self.tokenizer.save_pretrained(str(final_output_dir))

        logger.info(f"训练完成，耗时: {training_time:.2f}秒")
        logger.info(f"模型已保存到: {final_output_dir}")

        return TrainingResult(
            checkpoint_path=str(final_output_dir),
            final_loss=train_result.training_loss,
            training_time=training_time,
            metrics=train_result.metrics
        )

    def save_lora_weights(self, output_path: str):
        """仅保存 LoRA 权重"""
        if hasattr(self.model, "save_pretrained"):
            self.model.save_pretrained(output_path)
            self.tokenizer.save_pretrained(output_path)
            logger.info(f"LoRA 权重已保存到: {output_path}")


def load_lora_model(base_model_path: str, lora_path: str, device: str = "cuda"):
    """加载带 LoRA 的模型用于推理"""
    from transformers import AutoModelForCausalLM, AutoTokenizer
    from peft import PeftModel

    # 加载基础模型
    base_model = AutoModelForCausalLM.from_pretrained(
        base_model_path,
        torch_dtype=torch.float16,
        device_map="auto",
        trust_remote_code=True
    )

    # 加载 LoRA 适配器
    model = PeftModel.from_pretrained(base_model, lora_path)

    # 加载 tokenizer
    tokenizer = AutoTokenizer.from_pretrained(
        lora_path,
        trust_remote_code=True
    )

    return model, tokenizer


if __name__ == "__main__":
    print("SFTTrainerWrapper 模块")
```

```python
def setup_logging(log_file: Optional[str] = None, level: int = logging.INFO):
    """设置日志"""
    handlers = [logging.StreamHandler(sys.stdout)]

    if log_file:
        Path(log_file).parent.mkdir(parents=True, exist_ok=True)
        handlers.append(logging.FileHandler(log_file, encoding="utf-8"))

    logging.basicConfig(
        level=level,
        format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
        handlers=handlers
    )


def set_seed(seed: int):
    """设置随机种子"""
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    if torch.cuda.is_available():
        torch.cuda.manual_seed_all(seed)


def get_device() -> str:
    """获取可用设备"""
    if torch.cuda.is_available():
        return "cuda"
    elif torch.backends.mps.is_available():
        return "mps"
    else:
        return "cpu"


def get_gpu_memory_usage() -> Dict[str, float]:
    """获取 GPU 显存使用情况"""
    if not torch.cuda.is_available():
        return {"error": "CUDA not available"}

    result = {}
    for i in range(torch.cuda.device_count()):
        allocated = torch.cuda.memory_allocated(i) / 1024**3
        reserved = torch.cuda.memory_reserved(i) / 1024**3
        result[f"gpu_{i}"] = {
            "allocated_gb": round(allocated, 2),
            "reserved_gb": round(reserved, 2)
        }

    return result


def count_parameters(model) -> Dict[str, int]:
    """统计模型参数"""
    total = sum(p.numel() for p in model.parameters())
    trainable = sum(p.numel() for p in model.parameters() if p.requires_grad)

    return {
        "total": total,
        "trainable": trainable,
        "frozen": total - trainable,
        "trainable_ratio": trainable / total if total > 0 else 0
    }


def ensure_dir(path: str):
    """确保目录存在"""
    Path(path).mkdir(parents=True, exist_ok=True)


def format_time(seconds: float) -> str:
    """格式化时间"""
    if seconds < 60:
        return f"{seconds:.1f}s"
    elif seconds < 3600:
        minutes = seconds / 60
        return f"{minutes:.1f}min"
    else:
        hours = seconds / 3600
        return f"{hours:.2f}h"
```

```python
def generate_prediction(model, tokenizer, question: str, system_prompt: str = None):
    """生成预测"""
    messages = []

    if system_prompt:
        messages.append({"role": "system", "content": system_prompt})

    messages.append({"role": "user", "content": question})

    # 使用 ChatML 模板
    text = tokenizer.apply_chat_template(
        messages,
        tokenize=False,
        add_generation_prompt=True
    )

    inputs = tokenizer(text, return_tensors="pt").to(model.device)

    with torch.no_grad():
        outputs = model.generate(
            **inputs,
            max_new_tokens=512,
            temperature=0.1,
            top_p=0.9,
            do_sample=True,
            pad_token_id=tokenizer.pad_token_id,
            eos_token_id=tokenizer.eos_token_id
        )

    response = tokenizer.decode(outputs[0][inputs["input_ids"].shape[1]:], skip_special_tokens=True)
    return response.strip()
```
