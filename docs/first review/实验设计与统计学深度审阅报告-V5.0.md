#### 🔴 严重问题S4：缺乏多重比较校正

**位置**：
- 6.1.2节统计显著性检验（第859行）

**问题描述**：
```python
# 当前设计
配对t检验：p < 0.05
Wilcoxon检验：p < 0.05
# 但未提及多重比较校正
```

**科学问题**：
- 9个实验配置
- 至少8组两两比较（Exp3-8 vs Exp2）
- 在α=0.05水平下，**至少犯一次错的概率**：
  ```
  P(at least one error) = 1 - (1-0.05)^8 ≈ 34%
  ```

**这意味着**：即使所有组件都无效，仍有34%的概率得到至少一个"显著"结果

**修改建议**：

**方案A（推荐）**：Holm-Bonferroni校正
```python
from scipy.stats import ttest_ind

def holm_bonferroni(p_values, alpha=0.05):
    """
    Holm-Bonferroni校正步骤：
    1. 将p值从小到大排序：p1 ≤ p2 ≤ ... ≤ pk
    2. 从最小p值开始检验
    3. 如果 pi < alpha/(k-i+1)，则拒绝并继续；否则停止
    """
    k = len(p_values)
    sorted_indices = np.argsort(p_values)
    sorted_p = p_values[sorted_indices]

    for i, p in enumerate(sorted_p):
        threshold = alpha / (k - i)
        if p < threshold:
            continue  # 拒绝H0
        else:
            break  # 停止

    return sorted_indices[:i+1]  # 返回显著结果的索引

# 使用示例
p_values = np.array([0.001, 0.02, 0.03, 0.06, 0.10])
significant = holm_bonferroni(p_values, alpha=0.05)
```

**方案B**：Benjamini-Hochberg FDR控制
```python
def benjamini_hochberg(p_values, q=0.05):
    """FDR（False Discovery Rate）控制"""
    k = len(p_values)
    sorted_indices = np.argsort(p_values)
    sorted_p = p_values[sorted_indices]

    thresholds = (np.arange(1, k+1) / k) * q
    significant = sorted_p <= thresholds

    return sorted_indices[significant]
```

**方案C**：在论文中明确说明
```markdown

控制家族错误率（Family-wise Error Rate）。校正后的
显著性阈值范围为0.006（8次比较）至0.05（1次比较）。"
```

### 2.2 显著性检验方法

#### 🟡 中等问题M4：配对t检验的假设未验证

**位置**：
- 6.1.2节统计显著性检验（第859行）

**问题描述**：
```python
配对t检验：同一方法不同配置对比
Wilcoxon检验：非正态分布数据
```

**科学问题**：
1. **未说明何时使用哪种检验**：什么情况下用t检验？什么情况下用Wilcoxon？
2. **未说明如何检验正态性**：使用Shapiro-Wilk？还是看Q-Q图？
3. **小样本量问题**：n=3时，正态性检验功效极低

**修改建议**：
```python
# 推荐流程
def statistical_test(x, y, alpha=0.05):
    """
    显著性检验选择流程：
    1. 检查正态性（Shapiro-Wilk）
    2. 检查方差齐性（Levene's test）
    3. 选择合适的检验方法
    """
    from scipy import stats

    # 1. 正态性检验（两组数据分别检验）
    _, p_norm_x = stats.shapiro(x)
    _, p_norm_y = stats.shapiro(y)

    if p_norm_x > alpha and p_norm_y > alpha:
        # 2. 方差齐性检验
        _, p_levene = stats.levene(x, y)

        if p_levene > alpha:
            # 独立样本t检验（假设方差相等）
            stat, p_value = stats.ttest_ind(x, y, equal_var=True)
            test_type = "Student's t-test"
        else:
            # Welch's t检验（不假设方差相等）
            stat, p_value = stats.ttest_ind(x, y, equal_var=False)
            test_type = "Welch's t-test"
    else:
        # 非参数检验
        stat, p_value = stats.wilcoxon(x, y)
        test_type = "Wilcoxon signed-rank test"

    return p_value, test_type
```

**优先级**：P1（建议修改）

```python
# 完整的效应量报告
def effect_size(x, y, confidence=0.95):
    """计算多种效应量及其置信区间"""

    # 1. Cohen's d（参数）
    cohens_d = (np.mean(y) - np.mean(x)) / np.sqrt((np.std(x)**2 + np.std(y)**2) / 2)

    # Cohen's d的置信区间（非中心t分布）
    from scipy.stats import nct
    n1, n2 = len(x), len(y)
    se_d = np.sqrt((n1+n2)/(n1*n2) + cohens_d**2/(2*(n1+n2)))
    ci_low = cohens_d - 1.96 * se_d
    ci_high = cohens_d + 1.96 * se_d

    # 2. Cliff's Delta（非参数）
    # Cliff's Delta适用于非正态分布数据
    def cliffs_delta(x, y):
        nx, ny = len(x), len(y)
        dominances = 0
        for xi in x:
            for yi in y:
                if xi > yi:
                    dominances += 1
                elif xi < yi:
                    dominances -= 1
        return dominances / (nx * ny)

    cliffs_d = cliffs_delta(x, y)

    # 3. Glass's Delta（当标准差差异较大时）
    glass_delta = (np.mean(y) - np.mean(x)) / np.std(x)

    return {
        "cohens_d": (cohens_d, ci_low, ci_high),
        "cliffs_delta": cliffs_d,
        "glass_delta": glass_delta
    }
```

**论文中报告格式**：

```markdown
t(4) = 3.45, p = 0.026*，Cohen's d = 1.93 [0.23, 3.63]，Cliff's Delta = 0.67）。"
```

```python
# 在6.1.2节添加
def multiple_comparison_correction(p_values, method='holm'):
    """
    多重比较校正

    Args:
        p_values: 各组比较的p值
        method: 'holm' (推荐) 或 'bh' (FDR控制)

    Returns:
        corrected_p: 校正后的p值
        significant: 是否显著（α=0.05）
    """
    from scipy.stats import ttest_ind

    if method == 'holm':
        # Holm-Bonferroni校正
        k = len(p_values)
        sorted_indices = np.argsort(p_values)
        sorted_p = p_values[sorted_indices]

        for i in range(k):
            if sorted_p[i] < 0.05 / (k - i):
                continue
            else:
                # 从i开始的所有比较都不显著
                for j in range(i, k):
                    sorted_indices[j] = -1  # 标记为不显著
                break

        return sorted_indices

    elif method == 'bh':
        # Benjamini-Hochberg FDR控制
        k = len(p_values)
        sorted_indices = np.argsort(p_values)
        sorted_p = p_values[sorted_indices]
        thresholds = (np.arange(1, k+1) / k) * 0.05

        significant = sorted_p <= thresholds
        return sorted_indices[significant]
```

### 6.2 统计分析脚本优化

```python
# 在experiments/目录下创建统计分析脚本
# experiments/statistical_analysis.py

import numpy as np
from scipy import stats
from typing import Dict, List, Tuple

class StatisticalAnalyzer:
    """消融实验统计分析工具"""

    def __init__(self, alpha=0.05, n_comparisons=8):
        self.alpha = alpha
        self.n_comparisons = n_comparisons

    def compare_experiments(self,
                          exp_a_scores: List[float],
                          exp_b_scores: List[float],
                          exp_a_name: str,
                          exp_b_name: str) -> Dict:
        """
        比较两个实验的性能

        Returns:
            {
                'mean_a': float,
                'mean_b': float,
                'mean_diff': float,
                'p_value': float,
                'p_value_corrected': float,
                'cohens_d': float,
                'cohens_d_ci': Tuple[float, float],
                'cliffs_delta': float,
                'test_type': str,
                'is_significant': bool
            }
        """

        # 1. 描述性统计
        mean_a, std_a = np.mean(exp_a_scores), np.std(exp_a_scores, ddof=1)
        mean_b, std_b = np.mean(exp_b_scores), np.std(exp_b_scores, ddof=1)
        mean_diff = mean_b - mean_a

        # 2. 显著性检验
        # 检验正态性
        _, p_norm_a = stats.shapiro(exp_a_scores)
        _, p_norm_b = stats.shapiro(exp_b_scores)

        if p_norm_a > self.alpha and p_norm_b > self.alpha:
            # 参数检验
            # 检验方差齐性
            _, p_levene = stats.levene(exp_a_scores, exp_b_scores)

            if p_levene > self.alpha:
                # Student's t-test
                stat, p_value = stats.ttest_ind(exp_b_scores, exp_a_scores)
                test_type = "Student's t-test"
            else:
                # Welch's t-test
                stat, p_value = stats.ttest_ind(exp_b_scores, exp_a_scores,
                                               equal_var=False)
                test_type = "Welch's t-test"
        else:
            # 非参数检验
            stat, p_value = stats.wilcoxon(exp_b_scores, exp_a_scores)
            test_type = "Wilcoxon signed-rank test"

        # 3. 多重比较校正（Holm-Bonferroni）
        # 这里简化处理，实际应该在所有比较后统一校正
        p_value_corrected = min(p_value * self.n_comparisons, 1.0)

        # 4. 效应量
        # Cohen's d
        pooled_std = np.sqrt((std_a**2 + std_b**2) / 2)
        cohens_d = mean_diff / pooled_std

        # Cohen's d的置信区间
        n1, n2 = len(exp_a_scores), len(exp_b_scores)
        se_d = np.sqrt((n1+n2)/(n1*n2) + cohens_d**2/(2*(n1+n2)))
        ci_low = cohens_d - 1.96 * se_d
        ci_high = cohens_d + 1.96 * se_d

        # Cliff's Delta
        def cliffs_delta(x, y):
            nx, ny = len(x), len(y)
            dominances = 0
            for xi in x:
                for yi in y:
                    if xi > yi:
                        dominances += 1
                    elif xi < yi:
                        dominances -= 1
            return dominances / (nx * ny)

        cliffs_d = cliffs_delta(exp_b_scores, exp_a_scores)

        return {
            'mean_a': mean_a,
            'mean_b': mean_b,
            'std_a': std_a,
            'std_b': std_b,
            'mean_diff': mean_diff,
            'p_value': p_value,
            'p_value_corrected': p_value_corrected,
            'cohens_d': cohens_d,
            'cohens_d_ci': (ci_low, ci_high),
            'cliffs_delta': cliffs_d,
            'test_type': test_type,
            'is_significant': p_value_corrected < self.alpha
        }

    def generate_report(self,
                       comparisons: List[Dict],
                       output_path: str):
        """生成统计分析报告（Markdown格式）"""

        report = "# 消融实验统计分析报告\n\n"
        report += f"## 统计方法\n\n"
        report += f"- 显著性水平：α = {self.alpha}\n"
        report += f"- 多重比较校正：Holm-Bonferroni\n"
        report += f"- 比较次数：{self.n_comparisons}\n\n"

        report += "## 实验对比结果\n\n"
        report += "| 对比 | 均值差异 | p值 | 校正p值 | Cohen's d | Cliff's Δ | 检验方法 | 显著 |\n"
        report += "|------|---------|-----|--------|-----------|-----------|---------|------|\n"

        for comp in comparisons:
            sig_mark = "***" if comp['is_significant'] else ""
            report += (f"| {comp['exp_a']} vs {comp['exp_b']} "
                      f"| {comp['mean_diff']:+.2%} "
                      f"| {comp['p_value']:.4f} "
                      f"| {comp['p_value_corrected']:.4f} "
                      f"| {comp['cohens_d']:.2f} "
                      f"| {comp['cliffs_delta']:.2f} "
                      f"| {comp['test_type']} "
                      f"| {'是' if comp['is_significant'] else '否'}{sig_mark} |\n")

        report += "\n## 解释\n\n"
        report += "- Cohen's d: < 0.2 (小), 0.2-0.8 (中), > 0.8 (大)\n"
        report += "- Cliff's Δ: < 0.147 (可忽略), 0.147-0.33 (小), 0.33-0.474 (中), > 0.474 (大)\n"
        report += "- *** 表示经多重比较校正后显著 (p < 0.05)\n"

        with open(output_path, 'w', encoding='utf-8') as f:
            f.write(report)

        print(f"报告已保存至: {output_path}")
```
