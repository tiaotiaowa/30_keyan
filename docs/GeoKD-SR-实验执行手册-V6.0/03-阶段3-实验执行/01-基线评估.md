## 一、基线评估概述

### 1.1 评估目的

基线评估旨在：
1. 确立教师模型的性能上限
2. 了解学生模型的初始能力
3. 为蒸馏效果对比提供基准

### 1.2 评估指标

| 指标 | 说明 | 计算方法 |
|------|------|----------|
| Accuracy | 答题正确率 | 正确数/总数 |
| Exact Match | 完全匹配率 | 完全一致数/总数 |
| F1 Score | F1分数 | 2*Precision*Recall/(Precision+Recall) |
| Per-Category Acc | 分类别正确率 | 各类别正确率 |

### 1.3 评估数据集

- **主要数据集**：GeoSR-Bench-3000
- **评估方式**：批量推理
- **输出格式**：JSON结果文件

## 二、教师模型基线评估

### 2.1 评估脚本

创建 `scripts/eval_teacher_baseline.py`：

```python
#!/usr/bin/env python3
"""
教师模型基线评估脚本
评估Qwen2.5-7B-Instruct在GeoSR-Bench-3000上的性能
"""

import os
import json
import torch
from transformers import AutoTokenizer, AutoModelForCausalLM
from tqdm import tqdm
from datetime import datetime

def load_model(model_path):
    """加载教师模型"""
    print(f"加载教师模型: {model_path}")

    tokenizer = AutoTokenizer.from_pretrained(
        model_path,
        trust_remote_code=True
    )

    model = AutoModelForCausalLM.from_pretrained(
        model_path,
        torch_dtype=torch.float16,
        device_map="auto",
        trust_remote_code=True
    )

    model.eval()
    print(f"模型加载完成，设备: {model.device}")
    return model, tokenizer

def load_test_data(test_file):
    """加载测试数据"""
    print(f"加载测试数据: {test_file}")

    data = []
    with open(test_file, 'r', encoding='utf-8') as f:
        for line in f:
            data.append(json.loads(line))

    print(f"测试数据加载完成，共 {len(data)} 条")
    return data

def format_prompt(question, options=None):
    """格式化提示词"""
    prompt = f"请回答以下地理空间推理问题，并给出你的推理过程。\n\n问题：{question}\n\n"

    if options:
        prompt += "选项：\n"
        for key, value in options.items():
            prompt += f"{key}. {value}\n"
        prompt += "\n请选择正确选项，并说明理由。"

    return prompt

def generate_answer(model, tokenizer, prompt, max_new_tokens=512):
    """生成答案"""
    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)

    with torch.no_grad():
        outputs = model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            temperature=0.1,
            top_p=0.9,
            do_sample=True,
            pad_token_id=tokenizer.eos_token_id
        )

    response = tokenizer.decode(outputs[0][inputs['input_ids'].shape[1]:], skip_special_tokens=True)
    return response.strip()

def extract_answer(response):
    """从回答中提取答案选项"""
    # 简单实现：查找A/B/C/D
    import re
    match = re.search(r'[ABCD]', response.upper())
    return match.group(0) if match else None

def evaluate(model, tokenizer, test_data, output_file):
    """执行评估"""
    results = []
    correct = 0
    total = len(test_data)

    print(f"\n开始评估，共 {total} 题...")

    for item in tqdm(test_data, desc="评估进度"):
        prompt = format_prompt(item['question'], item.get('options'))

        try:
            response = generate_answer(model, tokenizer, prompt)
            predicted = extract_answer(response)

            result = {
                "id": item['id'],
                "question": item['question'],
                "ground_truth": item['answer'],
                "predicted": predicted,
                "response": response,
                "correct": predicted == item['answer']
            }

            if result['correct']:
                correct += 1

            results.append(result)

        except Exception as e:
            print(f"\n处理问题 {item['id']} 时出错: {e}")
            results.append({
                "id": item['id'],
                "error": str(e),
                "correct": False
            })

    # 计算准确率
    accuracy = correct / total
    print(f"\n教师模型准确率: {accuracy:.4f} ({correct}/{total})")

    # 保存结果
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump({
            "model": "Qwen2.5-7B-Instruct",
            "eval_date": datetime.now().isoformat(),
            "total": total,
            "correct": correct,
            "accuracy": accuracy,
            "results": results
        }, f, ensure_ascii=False, indent=2)

    print(f"结果已保存到: {output_file}")

    return accuracy, results

def main():
    """主函数"""
    # 配置
    TEACHER_MODEL = os.getenv("TEACHER_MODEL_PATH", "./models/teacher/Qwen2.5-7B-Instruct")
    TEST_DATA = os.getenv("TEST_DATA_PATH", "./data/test/GeoSR-Bench-3000.jsonl")
    OUTPUT_DIR = os.getenv("OUTPUT_DIR", "./outputs")
    OUTPUT_FILE = os.path.join(OUTPUT_DIR, "teacher_baseline_results.json")

    # 创建输出目录
    os.makedirs(OUTPUT_DIR, exist_ok=True)

    # 加载模型
    model, tokenizer = load_model(TEACHER_MODEL)

    # 加载数据
    test_data = load_test_data(TEST_DATA)

    # 评估
    accuracy, results = evaluate(model, tokenizer, test_data, OUTPUT_FILE)

    print("\n" + "="*50)
    print("教师模型基线评估完成")
    print(f"准确率: {accuracy:.4f}")
    print("="*50)

if __name__ == "__main__":
    main()
```

### 4.2 对比脚本

创建 `scripts/compare_baselines.py`：

```python
#!/usr/bin/env python3
"""
基线对比分析脚本
"""

import json
import os

def compare_baseline_results(teacher_file, student_file, output_file):
    """对比两个模型的基线结果"""

    # 加载结果
    with open(teacher_file, 'r', encoding='utf-8') as f:
        teacher_results = json.load(f)

    with open(student_file, 'r', encoding='utf-8') as f:
        student_results = json.load(f)

    # 对比分析
    comparison = {
        "teacher": {
            "model": teacher_results["model"],
            "accuracy": teacher_results["accuracy"],
            "correct": teacher_results["correct"],
            "total": teacher_results["total"]
        },
        "student": {
            "model": student_results["model"],
            "accuracy": student_results["accuracy"],
            "correct": student_results["correct"],
            "total": student_results["total"]
        },
        "gap": {
            "accuracy": teacher_results["accuracy"] - student_results["accuracy"],
            "correct": teacher_results["correct"] - student_results["correct"]
        },
        "improvement_needed": {
            "target_accuracy": teacher_results["accuracy"] * 0.95,  # 目标：达到教师模型95%性能
            "current_gap": teacher_results["accuracy"] - student_results["accuracy"]
        }
    }

    # 保存对比结果
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(comparison, f, ensure_ascii=False, indent=2)

    # 打印结果
    print("\n" + "="*60)
    print("基线对比分析")
    print("="*60)
    print(f"教师模型准确率: {comparison['teacher']['accuracy']:.4f}")
    print(f"学生模型准确率: {comparison['student']['accuracy']:.4f}")
    print(f"性能差距: {comparison['gap']['accuracy']:.4f}")
    print(f"目标准确率: {comparison['improvement_needed']['target_accuracy']:.4f}")
    print("="*60)

def main():
    OUTPUT_DIR = os.getenv("OUTPUT_DIR", "./outputs")

    teacher_file = os.path.join(OUTPUT_DIR, "teacher_baseline_results.json")
    student_file = os.path.join(OUTPUT_DIR, "student_baseline_results.json")
    output_file = os.path.join(OUTPUT_DIR, "baseline_comparison.json")

    compare_baseline_results(teacher_file, student_file, output_file)

if __name__ == "__main__":
    main()
```

## 七、常见问题排查

### 问题1：模型加载失败

**现象**：OSError或FileNotFoundError

**解决方案**：
1. 检查模型路径是否正确
2. 检查模型文件是否完整
3. 检查磁盘空间是否充足

### 问题2：推理OOM

**现象**：CUDA out of memory

**解决方案**：
1. 减小max_new_tokens
2. 使用更小的batch size
3. 使用8bit量化加载

### 问题3：答案解析失败

**现象**：extract_answer返回None

**解决方案**：
1. 改进prompt格式
2. 优化答案提取逻辑
3. 添加更多解析规则

### 问题4：评估速度慢

**现象**：评估时间过长

**解决方案**：
1. 使用批处理推理
2. 使用多GPU并行
3. 使用量化模型
