### 2.1 models/ - 模型定义

#### student_model.py
```python
"""
学生模型定义
支持不同规模的LLaMA架构变体
"""
from transformers import LlamaForCausalLM, LlamaConfig
from typing import Optional, Dict, Any

class StudentModel:
    """学生模型包装类"""

    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.model = self._build_model()

    def _build_model(self) -> LlamaForCausalLM:
        """构建学生模型"""
        config = LlamaConfig(
            hidden_size=self.config.get("hidden_size", 2048),
            num_attention_heads=self.config.get("num_heads", 32),
            num_hidden_layers=self.config.get("num_layers", 24),
            intermediate_size=self.config.get("intermediate_size", 5632),
            vocab_size=self.config.get("vocab_size", 128256),
        )
        return LlamaForCausalLM(config)

    def forward(self, input_ids, attention_mask=None, labels=None):
        """前向传播"""
        return self.model(
            input_ids=input_ids,
            attention_mask=attention_mask,
            labels=labels,
            output_hidden_states=True,
            return_dict=True
        )
```

#### teacher_model.py
```python
"""
教师模型定义
支持GLM-4和DeepSeek-V3两种教师模型
"""
from transformers import AutoModelForCausalLM, AutoTokenizer

class TeacherModel:
    """教师模型包装类"""

    def __init__(self, model_name: str, cache_dir: str = "./cache"):
        self.model_name = model_name
        self.tokenizer = AutoTokenizer.from_pretrained(
            model_name,
            trust_remote_code=True,
            cache_dir=cache_dir
        )
        self.model = AutoModelForCausalLM.from_pretrained(
            model_name,
            trust_remote_code=True,
            cache_dir=cache_dir,
            device_map="auto",
            torch_dtype="auto"
        )

    def generate_with_reasoning(self, prompt: str) -> Dict[str, Any]:
        """生成带推理的输出"""
        outputs = self.model.generate(
            **self.tokenizer(prompt, return_tensors="pt"),
            max_new_tokens=2048,
            temperature=0.7,
            do_sample=True,
            output_hidden_states=True,
            return_dict_in_generate=True
        )
        return {
            "logits": outputs.logits,
            "hidden_states": outputs.hidden_states,
            "text": self.tokenizer.decode(outputs.sequences[0])
        }
```

### 2.2 models/losses/ - 损失函数

#### distillation_loss.py
```python
"""
标准蒸馏损失 (B2基线)
基于Hinton 2015的KL散度蒸馏
"""
import torch
import torch.nn as nn
import torch.nn.functional as F

class StandardKDLoss(nn.Module):
    """
    标准知识蒸馏损失
    L_KD = α * L_soft + (1-α) * L_hard
    """

    def __init__(self, temperature: float = 2.0, alpha: float = 0.5):
        super().__init__()
        self.temperature = temperature
        self.alpha = alpha

    def forward(
        self,
        student_logits: torch.Tensor,
        teacher_logits: torch.Tensor,
        labels: torch.Tensor
    ) -> torch.Tensor:
        """
        计算蒸馏损失

        Args:
            student_logits: 学生模型输出 [batch_size, seq_len, vocab_size]
            teacher_logits: 教师模型输出 [batch_size, seq_len, vocab_size]
            labels: 真实标签 [batch_size, seq_len]
        """
        # 软标签损失 (KL散度)
        soft_loss = self._kl_div_loss(
            student_logits,
            teacher_logits
        )

        # 硬标签损失 (交叉熵)
        hard_loss = F.cross_entropy(
            student_logits.view(-1, student_logits.size(-1)),
            labels.view(-1),
            ignore_index=-100
        )

        return self.alpha * soft_loss + (1 - self.alpha) * hard_loss

    def _kl_div_loss(
        self,
        student_logits: torch.Tensor,
        teacher_logits: torch.Tensor
    ) -> torch.Tensor:
        """计算KL散度损失"""
        log_student = F.log_softmax(
            student_logits / self.temperature, dim=-1
        )
        soft_teacher = F.softmax(
            teacher_logits / self.temperature, dim=-1
        )

        return F.kl_div(
            log_student,
            soft_teacher,
            reduction="batchmean"
        ) * (self.temperature ** 2)
```

### 2.3 models/data/ - 数据处理

#### dataset.py
```python
"""
空间推理数据集
支持多种空间关系类型的标注数据
"""
from torch.utils.data import Dataset
from typing import Dict, List, Optional
import json

class SpatialReasoningDataset(Dataset):
    """
    空间推理数据集

    支持的空间关系类型:
    - directional: 方向关系 (东、南、西、北等)
    - topological: 拓扑关系 (相邻、包含、相交等)
    - metric: 度量关系 (距离比较)
    - hybrid: 组合关系
    """

    SPATIAL_RELATION_TYPES = [
        "directional",
        "topological",
        "metric",
        "hybrid"
    ]

    def __init__(
        self,
        data_path: str,
        tokenizer: Any,
        max_length: int = 2048,
        relation_filter: Optional[List[str]] = None
    ):
        self.data = self._load_data(data_path)
        self.tokenizer = tokenizer
        self.max_length = max_length
        self.relation_filter = relation_filter or self.SPATIAL_RELATION_TYPES

        # 按关系类型过滤
        if relation_filter:
            self.data = [
                d for d in self.data
                if d.get("relation_type") in relation_filter
            ]

    def _load_data(self, path: str) -> List[Dict]:
        """加载JSON格式数据"""
        with open(path, 'r', encoding='utf-8') as f:
            return json.load(f)

    def __len__(self) -> int:
        return len(self.data)

    def __getitem__(self, idx: int) -> Dict[str, torch.Tensor]:
        item = self.data[idx]

        # 构建输入文本
        prompt = self._format_prompt(item)

        # Tokenize
        encoded = self.tokenizer(
            prompt,
            max_length=self.max_length,
            padding="max_length",
            truncation=True,
            return_tensors="pt"
        )

        return {
            "input_ids": encoded["input_ids"].squeeze(0),
            "attention_mask": encoded["attention_mask"].squeeze(0),
            "labels": encoded["input_ids"].squeeze(0),
            "relation_type": item.get("relation_type", "unknown"),
            "question_id": item.get("id", idx)
        }

    def _format_prompt(self, item: Dict) -> str:
        """格式化提示词"""
        return f"""请分析以下空间关系问题：

问题：{item['question']}
选项：{item.get('options', '')}

请逐步分析并给出答案。"""
```

## 四、代码规范

### 4.1 命名规范

- **类名**: PascalCase (例: `StudentModel`)
- **函数名**: snake_case (例: `train_one_epoch`)
- **常量**: UPPER_SNAKE_CASE (例: `MAX_LENGTH`)
- **私有方法**: _leading_underscore (例: `_load_data`)

每个模块必须包含:
1. 模块级docstring说明功能
