```python
# models/data/dataset.py

from torch.utils.data import Dataset
from typing import Dict, List, Optional
import json

class GeoKDDataset(Dataset):
    """GeoKD-SR基础数据集类"""

    def __init__(
        self,
        data_path: str,
        tokenizer,
        max_length: int = 2048,
        split: str = "train"
    ):
        self.tokenizer = tokenizer
        self.max_length = max_length
        self.data = self._load_and_validate(data_path, split)

    def _load_and_validate(self, path: str, split: str) -> List[Dict]:
        """加载并验证数据格式"""
        with open(path, 'r', encoding='utf-8') as f:
            raw_data = json.load(f)

        # 验证必需字段
        validated = []
        for item in raw_data:
            if self._validate_item(item):
                validated.append(item)
        return validated

    def _validate_item(self, item: Dict) -> bool:
        """验证单条数据格式"""
        required_fields = ["id", "question", "answer", "relation_type"]
        return all(field in item for field in required_fields)

    def __len__(self) -> int:
        return len(self.data)

    def __getitem__(self, idx: int) -> Dict[str, torch.Tensor]:
        item = self.data[idx]
        prompt = self._format_prompt(item)

        encoded = self.tokenizer(
            prompt,
            max_length=self.max_length,
            padding="max_length",
            truncation=True,
            return_tensors="pt"
        )

        return {
            "input_ids": encoded["input_ids"].squeeze(0),
            "attention_mask": encoded["attention_mask"].squeeze(0),
            "labels": encoded["input_ids"].clone(),
            "relation_type": item["relation_type"],
            "metadata": {
                "id": item["id"],
                "answer": item["answer"]
            }
        }

    def _format_prompt(self, item: Dict) -> str:
        """格式化为标准提示格式"""
        return f"""问题：{item['question']}

请分析上述空间关系问题，并给出你的答案和推理过程。

答案：{item['answer']}"""
```

```python
from enum import Enum
from typing import Dict, List

class SpatialRelationType(Enum):
    """空间关系类型枚举"""
    DIRECTIONAL = "directional"      # 方向关系
    TOPOLOGICAL = "topological"      # 拓扑关系
    METRIC = "metric"                # 度量关系
    HYBRID = "hybrid"                # 组合关系

class SpatialRelationProcessor:
    """空间关系标签处理器"""

    # 关系类型权重（用于后续损失加权）
    RELATION_WEIGHTS = {
        SpatialRelationType.DIRECTIONAL: 1.0,
        SpatialRelationType.TOPOLOGICAL: 1.2,
        SpatialRelationType.METRIC: 1.1,
        SpatialRelationType.HYBRID: 1.5,
    }

    def __init__(self):
        self.relation_counts = {r: 0 for r in SpatialRelationType}

    def process_batch(
        self,
        relation_types: List[str]
    ) -> Dict[str, torch.Tensor]:
        """
        处理批次数据的关系类型

        Returns:
            {
                "type_ids": Tensor,     # 关系类型ID
                "weights": Tensor,      # 样本权重
                "mask": Tensor          # 关系类型mask
            }
        """
        type_ids = []
        weights = []

        for rt in relation_types:
            enum_type = SpatialRelationType(rt)
            type_ids.append(enum_type.value)
            weights.append(self.RELATION_WEIGHTS[enum_type])
            self.relation_counts[enum_type] += 1

        return {
            "type_ids": torch.tensor(type_ids),
            "weights": torch.tensor(weights),
        }

    def get_statistics(self) -> Dict[str, int]:
        """获取数据集统计信息"""
        return {r.name: c for r, c in self.relation_counts.items()}
```

```python
# scripts/validate_data.py

import json
from pathlib import Path
from typing import Dict, List, Tuple

class DataValidator:
    """数据质量验证器"""

    REQUIRED_FIELDS = ["id", "question", "answer", "relation_type"]
    VALID_RELATIONS = ["directional", "topological", "metric", "hybrid"]

    def __init__(self, data_path: str):
        self.data_path = Path(data_path)
        self.errors = []
        self.warnings = []
        self.stats = {
            "total": 0,
            "valid": 0,
            "by_relation": {r: 0 for r in self.VALID_RELATIONS}
        }

    def validate(self) -> Tuple[bool, Dict]:
        """执行完整验证"""
        with open(self.data_path, 'r', encoding='utf-8') as f:
            data = json.load(f)

        self.stats["total"] = len(data)

        for idx, item in enumerate(data):
            self._validate_item(item, idx)

        self._generate_report()
        return len(self.errors) == 0, self.stats

    def _validate_item(self, item: Dict, idx: int):
        """验证单条数据"""
        # 检查必需字段
        for field in self.REQUIRED_FIELDS:
            if field not in item:
                self.errors.append(f"Line {idx}: Missing field '{field}'")
                return

        # 检查关系类型
        if item["relation_type"] not in self.VALID_RELATIONS:
            self.errors.append(
                f"Line {idx}: Invalid relation_type '{item['relation_type']}'"
            )
            return

        # 检查问题长度
        if len(item["question"]) < 10:
            self.warnings.append(
                f"Line {idx}: Question too short ({len(item['question'])} chars)"
            )

        # 统计
        self.stats["valid"] += 1
        self.stats["by_relation"][item["relation_type"]] += 1

    def _generate_report(self):
        """生成验证报告"""
        print("=" * 50)
        print("数据验证报告")
        print("=" * 50)
        print(f"总数据量: {self.stats['total']}")
        print(f"有效数据: {self.stats['valid']}")
        print(f"错误数: {len(self.errors)}")
        print(f"警告数: {len(self.warnings)}")
        print("\n按关系类型分布:")
        for rel, count in self.stats["by_relation"].items():
            print(f"  {rel}: {count}")

        if self.errors:
            print(f"\n错误详情:")
            for err in self.errors[:10]:
                print(f"  - {err}")
            if len(self.errors) > 10:
                print(f"  ... 还有 {len(self.errors) - 10} 个错误")

if __name__ == "__main__":
    import sys
    validator = DataValidator(sys.argv[1])
    is_valid, stats = validator.validate()
    sys.exit(0 if is_valid else 1)
```

```python
# experiments/trainer.py

from typing import Dict, Optional, Callable
import torch
import torch.nn as nn
from torch.utils.data import DataLoader
from tqdm import tqdm
import wandb

class BaseTrainer:
    """基础训练器"""

    def __init__(
        self,
        model: nn.Module,
        optimizer: torch.optim.Optimizer,
        scheduler: Optional[torch.optim.lr_scheduler._LRScheduler] = None,
        device: str = "cuda",
        wandb_project: Optional[str] = None
    ):
        self.model = model.to(device)
        self.optimizer = optimizer
        self.scheduler = scheduler
        self.device = device
        self.global_step = 0

        if wandb_project:
            wandb.init(project=wandb_project)

    def train_epoch(
        self,
        train_loader: DataLoader,
        loss_fn: Callable,
        epoch: int
    ) -> Dict[str, float]:
        """训练一个epoch"""
        self.model.train()
        total_loss = 0
        num_batches = 0

        pbar = tqdm(train_loader, desc=f"Epoch {epoch}")
        for batch in pbar:
            loss = self.train_step(batch, loss_fn)
            total_loss += loss.item()
            num_batches += 1

            # 日志记录
            if self.global_step % 10 == 0:
                wandb.log({"train/loss": loss.item()}, step=self.global_step)

            pbar.set_postfix({"loss": loss.item()})
            self.global_step += 1

        return {"loss": total_loss / num_batches}

    def train_step(self, batch: Dict, loss_fn: Callable) -> torch.Tensor:
        """单步训练"""
        # 移动数据到设备
        batch = {k: v.to(self.device) for k, v in batch.items()
                 if isinstance(v, torch.Tensor)}

        # 前向传播
        outputs = self.model(**batch)
        loss = loss_fn(outputs, batch)

        # 反向传播
        self.optimizer.zero_grad()
        loss.backward()
        self.optimizer.step()

        if self.scheduler:
            self.scheduler.step()

        return loss

    def save_checkpoint(self, path: str, epoch: int):
        """保存检查点"""
        torch.save({
            "epoch": epoch,
            "model_state_dict": self.model.state_dict(),
            "optimizer_state_dict": self.optimizer.state_dict(),
            "global_step": self.global_step
        }, path)

    def load_checkpoint(self, path: str):
        """加载检查点"""
        checkpoint = torch.load(path)
        self.model.load_state_dict(checkpoint["model_state_dict"])
        self.optimizer.load_state_dict(checkpoint["optimizer_state_dict"])
        self.global_step = checkpoint["global_step"]
        return checkpoint["epoch"]
```

```python
# models/losses/sft_loss.py

import torch
import torch.nn as nn
import torch.nn.functional as F

class SFTLoss(nn.Module):
    """
    直接监督微调损失 (B1基线)
    L = CrossEntropy(logits, labels)
    """

    def forward(
        self,
        student_logits: torch.Tensor,
        labels: torch.Tensor,
        attention_mask: Optional[torch.Tensor] = None
    ) -> torch.Tensor:
        """
        Args:
            student_logits: [batch_size, seq_len, vocab_size]
            labels: [batch_size, seq_len]
            attention_mask: [batch_size, seq_len]
        """
        # 移动标签到设备
        labels = labels.to(student_logits.device)

        # 计算交叉熵
        loss = F.cross_entropy(
            student_logits.view(-1, student_logits.size(-1)),
            labels.view(-1),
            ignore_index=-100,  # 忽略padding
            reduction="mean"
        )

        return loss
```

```python
# models/losses/standard_kd_loss.py

import torch
import torch.nn as nn
import torch.nn.functional as F

class StandardKDLoss(nn.Module):
    """
    标准知识蒸馏损失 (B2基线)
    基于Hinton 2015

    L = α * L_soft + (1-α) * L_hard
    L_soft = T² * KL(Softmax(z_s/T) || Softmax(z_t/T))
    L_hard = CrossEntropy(z_s, y)
    """

    def __init__(
        self,
        temperature: float = 2.0,
        alpha: float = 0.5
    ):
        super().__init__()
        self.temperature = temperature
        self.alpha = alpha

    def forward(
        self,
        student_logits: torch.Tensor,
        teacher_logits: torch.Tensor,
        labels: torch.Tensor,
        attention_mask: Optional[torch.Tensor] = None
    ) -> torch.Tensor:
        """
        Args:
            student_logits: [batch_size, seq_len, vocab_size]
            teacher_logits: [batch_size, seq_len, vocab_size]
            labels: [batch_size, seq_len]
        """
        # 硬标签损失
        hard_loss = F.cross_entropy(
            student_logits.view(-1, student_logits.size(-1)),
            labels.view(-1),
            ignore_index=-100,
            reduction="mean"
        )

        # 软标签损失 (KL散度)
        log_student = F.log_softmax(
            student_logits / self.temperature, dim=-1
        )
        soft_teacher = F.softmax(
            teacher_logits / self.temperature, dim=-1
        )

        soft_loss = F.kl_div(
            log_student,
            soft_teacher,
            reduction="batchmean"
        ) * (self.temperature ** 2)

        # 组合
        total_loss = self.alpha * soft_loss + (1 - self.alpha) * hard_loss

        return total_loss
```

```python
# models/losses/spatial_relation_loss.py

import torch
import torch.nn as nn
import torch.nn.functional as F
from typing import Dict, Optional

class SpatialRelationDistillationLoss(nn.Module):
    """
    C1: 空间关系感知蒸馏损失

    根据空间关系类型动态调整蒸馏权重:

    L_total = Σᵢ w_{rt(i)} × L_KD(x_i)

    其中:
    - rt(i) ∈ {directional, topological, metric, hybrid}
    - w_{rt} 是关系类型的权重
    """

    # 默认权重配置（可通过消融实验调整）
    DEFAULT_WEIGHTS = {
        "directional": 1.0,
        "topological": 1.2,
        "metric": 1.1,
        "hybrid": 1.5,
    }

    def __init__(
        self,
        weights: Optional[Dict[str, float]] = None,
        temperature: float = 2.0,
        alpha: float = 0.5
    ):
        super().__init__()
        self.weights = weights or self.DEFAULT_WEIGHTS
        self.temperature = temperature
        self.alpha = alpha

        # 注册为buffer以便保存到模型
        self.register_buffer(
            "weight_tensor",
            torch.tensor(list(self.weights.values()))
        )

    def forward(
        self,
        student_logits: torch.Tensor,
        teacher_logits: torch.Tensor,
        labels: torch.Tensor,
        relation_types: torch.Tensor,  # [batch_size] 关系类型ID
        attention_mask: Optional[torch.Tensor] = None
    ) -> torch.Tensor:
        """
        Args:
            student_logits: [batch_size, seq_len, vocab_size]
            teacher_logits: [batch_size, seq_len, vocab_size]
            labels: [batch_size, seq_len]
            relation_types: [batch_size] 关系类型ID (0-3)
        """
        batch_size = student_logits.size(0)

        # 基础蒸馏损失
        base_loss = self._compute_kd_loss(
            student_logits,
            teacher_logits,
            labels
        )

        # 获取每个样本的权重
        sample_weights = self.weight_tensor[relation_types]  # [batch_size]

        # 加权损失
        weighted_loss = (base_loss * sample_weights).mean()

        return weighted_loss

    def _compute_kd_loss(
        self,
        student_logits: torch.Tensor,
        teacher_logits: torch.Tensor,
        labels: torch.Tensor
    ) -> torch.Tensor:
        """计算基础KD损失（按样本）"""
        # 软标签损失
        log_student = F.log_softmax(
            student_logits / self.temperature, dim=-1
        )
        soft_teacher = F.softmax(
            teacher_logits / self.temperature, dim=-1
        )

        # 按样本计算KL散度
        soft_loss = F.kl_div(
            log_student,
            soft_teacher,
            reduction="none"
        ).sum(dim=-1)  # [batch_size, seq_len]
        soft_loss = soft_loss.mean(dim=1) * (self.temperature ** 2)  # [batch_size]

        # 硬标签损失
        hard_loss = F.cross_entropy(
            student_logits.view(-1, student_logits.size(-1)),
            labels.view(-1),
            ignore_index=-100,
            reduction="none"
        )
        hard_loss = hard_loss.view(student_logits.size(0), -1).mean(dim=1)

        # 组合
        base_loss = self.alpha * soft_loss + (1 - self.alpha) * hard_loss

        return base_loss
```

```python
# models/losses/cot_distillation_loss.py

import torch
import torch.nn as nn
import torch.nn.functional as F
from typing import List, Dict, Optional

class CoTDistillationLoss(nn.Module):
    """
    C2: 思维链蒸馏损失

    通过蒸馏中间层表示，让学生模型学习教师模型的推理过程。

    L_CoT = L_final + λ × Σₗ L_layer(h_s^l, h_t^l)

    其中:
    - h_s^l: 学生模型第l层的隐藏状态
    - h_t^l: 教师模型第l层的隐藏状态
    - L_layer: 表示蒸馏损失（MSE或余弦相似度）
    """

    def __init__(
        self,
        layer_indices: List[int] = None,  # 要蒸馏的层索引
        lambda_cot: float = 0.5,
        loss_type: str = "mse"  # "mse" or "cosine"
    ):
        super().__init__()
        self.layer_indices = layer_indices or [4, 8, 12, 16, 20]  # 默认层
        self.lambda_cot = lambda_cot
        self.loss_type = loss_type

    def forward(
        self,
        student_outputs: Dict,
        teacher_outputs: Dict,
        labels: torch.Tensor
    ) -> torch.Tensor:
        """
        Args:
            student_outputs: {"logits": ..., "hidden_states": [...]}
            teacher_outputs: {"logits": ..., "hidden_states": [...]}
            labels: [batch_size, seq_len]
        """
        # 最终层损失
        final_loss = F.cross_entropy(
            student_outputs["logits"].view(-1, student_outputs["logits"].size(-1)),
            labels.view(-1),
            ignore_index=-100
        )

        # 中间层蒸馏损失
        cot_loss = self._compute_layer_distillation(
            student_outputs["hidden_states"],
            teacher_outputs["hidden_states"]
        )

        # 组合
        total_loss = final_loss + self.lambda_cot * cot_loss

        return total_loss

    def _compute_layer_distillation(
        self,
        student_hidden: List[torch.Tensor],
        teacher_hidden: List[torch.Tensor]
    ) -> torch.Tensor:
        """
        计算层间蒸馏损失

        Args:
            student_hidden: 学生模型所有层的隐藏状态
            teacher_hidden: 教师模型所有层的隐藏状态
        """
        layer_losses = []

        # 学生和教师模型的层数可能不同，需要映射
        num_student_layers = len(student_hidden)
        num_teacher_layers = len(teacher_hidden)

        for student_idx in self.layer_indices:
            # 映射到教师模型对应层
            teacher_idx = int(student_idx * num_teacher_layers / num_student_layers)

            # 获取隐藏状态
            s_hidden = student_hidden[student_idx]  # [B, S, D]
            t_hidden = teacher_hidden[teacher_idx]  # [B, S, D_t]

            # 维度对齐（教师维度可能更大）
            if s_hidden.size(-1) != t_hidden.size(-1):
                # 投影到相同维度
                if not hasattr(self, 'projection'):
                    self.projection = nn.Linear(
                        t_hidden.size(-1),
                        s_hidden.size(-1)
                    ).to(s_hidden.device)
                t_hidden = self.projection(t_hidden)

            # 计算损失
            if self.loss_type == "mse":
                loss = F.mse_loss(s_hidden, t_hidden)
            else:  # cosine
                loss = 1 - F.cosine_similarity(
                    s_hidden.view(-1, s_hidden.size(-1)),
                    t_hidden.view(-1, t_hidden.size(-1)),
                    dim=-1
                ).mean()

            layer_losses.append(loss)

        return torch.stack(layer_losses).mean()
```

```python
# models/losses/hybrid_kl_loss.py

import torch
import torch.nn as nn
import torch.nn.functional as F

class HybridKLDistillationLoss(nn.Module):
    """
    C3: 混合KL蒸馏损失

    结合Forward KL和Reverse KL的优势:

    L_hybrid = α(t) × KL(P||Q) + (1-α(t)) × KL(Q||P)

    其中:
    - KL(P||Q): Forward KL (鼓励学生覆盖教师支持)
    - KL(Q||P): Reverse KL (鼓励学生集中在教师高概率区)
    - α(t): 时间衰减权重
    """

    def __init__(
        self,
        temperature: float = 2.0,
        alpha_init: float = 0.5,
        alpha_final: float = 0.2,
        total_epochs: int = 10
    ):
        super().__init__()
        self.temperature = temperature
        self.alpha_init = alpha_init
        self.alpha_final = alpha_final
        self.total_epochs = total_epochs
        self.current_epoch = 0

    def set_epoch(self, epoch: int):
        """设置当前训练epoch"""
        self.current_epoch = epoch

    def get_alpha(self) -> float:
        """
        获取当前epoch的α权重

        使用线性衰减: α(t) = α_init - (α_init - α_final) × t/T
        """
        progress = self.current_epoch / self.total_epochs
        return self.alpha_init - (self.alpha_init - self.alpha_final) * progress

    def forward(
        self,
        student_logits: torch.Tensor,
        teacher_logits: torch.Tensor,
        labels: torch.Tensor
    ) -> torch.Tensor:
        """
        Args:
            student_logits: [batch_size, seq_len, vocab_size]
            teacher_logits: [batch_size, seq_len, vocab_size]
            labels: [batch_size, seq_len]
        """
        # 计算分布
        log_p = F.log_softmax(student_logits / self.temperature, dim=-1)
        log_q = F.log_softmax(teacher_logits / self.temperature, dim=-1)
        p = torch.exp(log_p)
        q = torch.exp(log_q)

        # Forward KL: KL(Q||P) = Σ q × (log q - log p)
        forward_kl = (q * (log_q - log_p)).sum(dim=-1).mean()

        # Reverse KL: KL(P||Q) = Σ p × (log p - log q)
        reverse_kl = (p * (log_p - log_q)).sum(dim=-1).mean()

        # 混合权重
        alpha = self.get_alpha()

        # 混合损失
        soft_loss = alpha * forward_kl + (1 - alpha) * reverse_kl
        soft_loss = soft_loss * (self.temperature ** 2)

        # 硬标签损失
        hard_loss = F.cross_entropy(
            student_logits.view(-1, student_logits.size(-1)),
            labels.view(-1),
            ignore_index=-100
        )

        # 组合
        total_loss = 0.5 * soft_loss + 0.5 * hard_loss

        return total_loss
```

```python
class HierarchicalWeightLoss(nn.Module):

    C4: 层级蒸馏权重

        self.weights = torch.linspace(min_weight, max_weight, num_layers)

    def get_layer_weight(self, layer_idx: int) -> float:
```

```python
# models/losses/attention_momentum_loss.py

class AttentionMomentumLoss(nn.Module):
    """
    C5: 注意力力矩蒸馏

    蒸馏注意力图和注意力力矩:

    L_attn = L_attn_map + λ × L_attn_momentum
    """

    def __init__(self, lambda_momentum: float = 0.3):
        super().__init__()
        self.lambda_momentum = lambda_momentum

    def forward(
        self,
        student_attn: torch.Tensor,
        teacher_attn: torch.Tensor
    ) -> torch.Tensor:
        # 注意力图损失
        map_loss = F.mse_loss(student_attn, teacher_attn)

        # 注意力力矩损失（二阶矩）
        student_moment = (student_attn ** 2).mean(dim=-1)
        teacher_moment = (teacher_attn ** 2).mean(dim=-1)
        momentum_loss = F.mse_loss(student_moment, teacher_moment)

        return map_loss + self.lambda_momentum * momentum_loss
```

```python
# models/losses/adaptive_sample_weight.py

class AdaptiveSampleWeightLoss(nn.Module):
    """
    C6: 自适应样本重加权

    根据蒸馏难度动态调整样本权重:

    w_i = f(difficulty_i)
    difficulty_i = ||p_s - p_t||

    难样本获得更高权重
    """

    def __init__(self, temperature: float = 2.0):
        super().__init__()
        self.temperature = temperature

    def forward(
        self,
        student_logits: torch.Tensor,
        teacher_logits: torch.Tensor,
        labels: torch.Tensor
    ) -> torch.Tensor:
        # 计算难度
        p_s = F.softmax(student_logits / self.temperature, dim=-1)
        p_t = F.softmax(teacher_logits / self.temperature, dim=-1)
        difficulty = (p_s - p_t).abs().sum(dim=-1)  # [batch_size]

        # 归一化难度作为权重
        weights = difficulty / difficulty.mean()

        # 加权损失
        base_loss = F.cross_entropy(
            student_logits.view(-1, student_logits.size(-1)),
            labels.view(-1),
            ignore_index=-100,
            reduction="none"
        )

        weighted_loss = (base_loss * weights.view(-1)).mean()

        return weighted_loss
```

```python
# experiments/evaluate.py

import argparse
import yaml
import torch
from pathlib import Path
from tqdm import tqdm
import json

def evaluate_model(
    model_path: str,
    data_path: str,
    config_path: str,
    device: str = "cuda"
) -> Dict:
    """评估单个模型"""
    # 加载配置
    with open(config_path) as f:
        config = yaml.safe_load(f)

    # 加载模型
    from models.student_model import StudentModel
    model = StudentModel(config["model"])
    model.load_state_dict(torch.load(model_path)["model_state_dict"])
    model = model.to(device)
    model.eval()

    # 加载数据
    from transformers import AutoTokenizer
    from models.data.dataset import GeoKDDataset
    from torch.utils.data import DataLoader

    tokenizer = AutoTokenizer.from_pretrained(config["model"]["name_or_path"])
    dataset = GeoKDDataset(data_path, tokenizer)
    loader = DataLoader(dataset, batch_size=8, shuffle=False)

    # 评估
    correct = 0
    total = 0
    results = []

    with torch.no_grad():
        for batch in tqdm(loader):
            batch = {k: v.to(device) for k, v in batch.items()
                     if isinstance(v, torch.Tensor)}

            outputs = model.generate(
                input_ids=batch["input_ids"],
                attention_mask=batch["attention_mask"],
                max_new_tokens=256
            )

            # 计算准确率
            predictions = tokenizer.batch_decode(outputs, skip_special_tokens=True)
            for pred, ref in zip(predictions, batch["metadata"]["answer"]):
                if ref in pred:
                    correct += 1
                total += 1

                results.append({
                    "prediction": pred,
                    "reference": ref,
                    "correct": ref in pred
                })

    metrics = {
        "accuracy": correct / total if total > 0 else 0,
        "total": total,
        "correct": correct
    }

    return metrics, results

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--checkpoint", type=str, required=True)
    parser.add_argument("--data", type=str, required=True)
    parser.add_argument("--config", type=str, required=True)
    parser.add_argument("--output", type=str, default="outputs/metrics/")
    args = parser.parse_args()

    metrics, results = evaluate_model(
        args.checkpoint,
        args.data,
        args.config
    )

    # 保存结果
    output_path = Path(args.output)
    output_path.mkdir(parents=True, exist_ok=True)

    with open(output_path / "metrics.json", 'w') as f:
        json.dump(metrics, f, indent=2)

    with open(output_path / "predictions.json", 'w') as f:
        json.dump(results, f, indent=2, ensure_ascii=False)

    print(f"Accuracy: {metrics['accuracy']:.4f} ({metrics['correct']}/{metrics['total']})")
    print(f"Results saved to {output_path}")

if __name__ == "__main__":
    main()
```
