## 一、指标体系概述

### 1.1 核心指标

| 指标类别 | 指标名称 | 说明 | 优先级 |
|----------|----------|------|--------|
| 准确性 | Accuracy | 答题正确率 | P0 |
| 准确性 | Exact Match | 完全匹配率 | P0 |
| 准确性 | F1 Score | F1分数 | P1 |
| 效率 | Inference Time | 推理耗时 | P1 |
| 效率 | Throughput | 吞吐量 | P2 |
| 资源 | Memory Usage | 显存占用 | P1 |
| 蒸馏效果 | Performance Gap | 性能差距 | P0 |
| 蒸馏效果 | Retention Ratio | 性能保留率 | P0 |

### 1.2 分级指标

**全局指标**：
- 总体准确率
- 平均推理时间
- 总体F1分数

**类别指标**：
- 按难度分类（简单/中等/困难）
- 按题型分类（空间关系/方位判断/距离计算等）
- 按推理步骤分类（单步/多步）

**样本级指标**：
- 单题正确率
- 单题推理时间
- 答案置信度

## 二、指标计算实现

### 2.1 基础指标计算

创建 `experiments/metrics/calculate_metrics.py`：

```python
#!/usr/bin/env python3
"""
基础指标计算模块
"""

import json
import numpy as np
from typing import List, Dict, Any
from collections import defaultdict

class MetricsCalculator:
    """指标计算器"""

    def __init__(self):
        self.results = []

    def load_results(self, result_file: str):
        """加载评估结果"""
        with open(result_file, 'r', encoding='utf-8') as f:
            data = json.load(f)

        self.results = data.get('results', [])
        return len(self.results)

    def calculate_accuracy(self) -> Dict[str, float]:
        """计算准确率"""
        total = len(self.results)
        if total == 0:
            return {"accuracy": 0.0, "correct": 0, "total": 0}

        correct = sum(1 for r in self.results if r.get('correct', False))

        return {
            "accuracy": correct / total,
            "correct": correct,
            "total": total
        }

    def calculate_per_category_accuracy(self) -> Dict[str, Dict[str, float]]:
        """计算类别准确率"""
        category_stats = defaultdict(lambda: {"correct": 0, "total": 0})

        for result in self.results:
            if 'category' in result:
                category = result['category']
                category_stats[category]["total"] += 1
                if result.get('correct', False):
                    category_stats[category]["correct"] += 1

        # 计算每个类别的准确率
        category_accuracy = {}
        for category, stats in category_stats.items():
            category_accuracy[category] = {
                "accuracy": stats["correct"] / stats["total"] if stats["total"] > 0 else 0,
                "correct": stats["correct"],
                "total": stats["total"]
            }

        return category_accuracy

    def calculate_per_difficulty_accuracy(self) -> Dict[str, Dict[str, float]]:
        """计算难度级别准确率"""
        difficulty_stats = defaultdict(lambda: {"correct": 0, "total": 0})

        for result in self.results:
            if 'difficulty' in result:
                difficulty = result['difficulty']
                difficulty_stats[difficulty]["total"] += 1
                if result.get('correct', False):
                    difficulty_stats[difficulty]["correct"] += 1

        difficulty_accuracy = {}
        for difficulty, stats in difficulty_stats.items():
            difficulty_accuracy[difficulty] = {
                "accuracy": stats["correct"] / stats["total"] if stats["total"] > 0 else 0,
                "correct": stats["correct"],
                "total": stats["total"]
            }

        return difficulty_accuracy

    def calculate_f1_score(self) -> Dict[str, float]:
        """计算F1分数（适用于多分类任务）"""
        # 收集所有预测和真实标签
        y_true = []
        y_pred = []

        for result in self.results:
            if 'ground_truth' in result and 'predicted' in result:
                y_true.append(result['ground_truth'])
                y_pred.append(result.get('predicted', ''))

        if not y_true:
            return {"f1_macro": 0.0, "f1_micro": 0.0}

        # 计算每个类别的precision和recall
        from sklearn.metrics import f1_score

        f1_macro = f1_score(y_true, y_pred, average='macro')
        f1_micro = f1_score(y_true, y_pred, average='micro')

        return {
            "f1_macro": f1_macro,
            "f1_micro": f1_micro
        }

    def calculate_inference_time_stats(self) -> Dict[str, float]:
        """计算推理时间统计"""
        times = [r.get('inference_time', 0) for r in self.results if r.get('inference_time')]

        if not times:
            return {
                "mean": 0,
                "median": 0,
                "min": 0,
                "max": 0,
                "std": 0
            }

        return {
            "mean": np.mean(times),
            "median": np.median(times),
            "min": np.min(times),
            "max": np.max(times),
            "std": np.std(times)
        }

    def calculate_all_metrics(self) -> Dict[str, Any]:
        """计算所有指标"""
        metrics = {
            "overall": self.calculate_accuracy(),
            "per_category": self.calculate_per_category_accuracy(),
            "per_difficulty": self.calculate_per_difficulty_accuracy(),
            "f1_score": self.calculate_f1_score(),
            "inference_time": self.calculate_inference_time_stats()
        }

        return metrics

    def save_metrics(self, output_file: str):
        """保存指标到文件"""
        metrics = self.calculate_all_metrics()

        with open(output_file, 'w', encoding='utf-8') as f:
            json.dump(metrics, f, ensure_ascii=False, indent=2)

        return metrics

def main():
    """主函数"""
    import os

    calculator = MetricsCalculator()

    # 加载结果文件
    result_file = "./outputs/student_baseline_results.json"
    calculator.load_results(result_file)

    # 计算并保存指标
    output_file = "./outputs/metrics/student_metrics.json"
    metrics = calculator.save_metrics(output_file)

    # 打印关键指标
    print("\n" + "="*50)
    print("指标计算结果")
    print("="*50)
    print(f"总体准确率: {metrics['overall']['accuracy']:.4f}")
    print(f"F1-Macro: {metrics['f1_score']['f1_macro']:.4f}")
    print(f"平均推理时间: {metrics['inference_time']['mean']:.2f}秒")
    print("="*50)

    # 打印类别准确率
    print("\n类别准确率:")
    for category, stats in metrics['per_category'].items():
        print(f"  {category}: {stats['accuracy']:.4f} ({stats['correct']}/{stats['total']})")

if __name__ == "__main__":
    main()
```

### 2.2 蒸馏效果指标

创建 `experiments/metrics/calculate_distillation_metrics.py`：

```python
#!/usr/bin/env python3
"""
蒸馏效果指标计算
"""

import json
import os
from typing import Dict, Any

def calculate_distillation_metrics(
    teacher_file: str,
    student_before_file: str,
    student_after_file: str,
    output_file: str
):
    """
    计算蒸馏效果指标

    Args:
        teacher_file: 教师模型结果文件
        student_before_file: 蒸馏前学生模型结果文件
        student_after_file: 蒸馏后学生模型结果文件
        output_file: 输出文件
    """

    # 加载结果
    with open(teacher_file, 'r') as f:
        teacher_data = json.load(f)

    with open(student_before_file, 'r') as f:
        student_before_data = json.load(f)

    with open(student_after_file, 'r') as f:
        student_after_data = json.load(f)

    # 提取准确率
    teacher_acc = teacher_data['accuracy']
    student_before_acc = student_before_data['accuracy']
    student_after_acc = student_after_data['accuracy']

    # 计算蒸馏效果指标
    metrics = {
        "baseline": {
            "teacher_accuracy": teacher_acc,
            "student_before_accuracy": student_before_acc,
            "initial_gap": teacher_acc - student_before_acc
        },
        "after_distillation": {
            "student_after_accuracy": student_after_acc,
            "final_gap": teacher_acc - student_after_acc
        },
        "improvement": {
            "absolute_improvement": student_after_acc - student_before_acc,
            "relative_improvement": (student_after_acc - student_before_acc) / (teacher_acc - student_before_acc) if teacher_acc != student_before_acc else 0,
            "performance_retention": student_after_acc / teacher_acc
        },
        "target": {
            "target_accuracy": teacher_acc * 0.95,
            "target_achieved": student_after_acc >= teacher_acc * 0.95
        }
    }

    # 保存结果
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(metrics, f, ensure_ascii=False, indent=2)

    # 打印结果
    print("\n" + "="*60)
    print("蒸馏效果分析")
    print("="*60)
    print(f"教师模型准确率: {metrics['baseline']['teacher_accuracy']:.4f}")
    print(f"学生模型蒸馏前: {metrics['baseline']['student_before_accuracy']:.4f}")
    print(f"学生模型蒸馏后: {metrics['after_distillation']['student_after_accuracy']:.4f}")
    print(f"\n绝对提升: {metrics['improvement']['absolute_improvement']:.4f}")
    print(f"相对提升: {metrics['improvement']['relative_improvement']:.2%}")
    print(f"性能保留率: {metrics['improvement']['performance_retention']:.2%}")
    print(f"\n目标准确率: {metrics['target']['target_accuracy']:.4f}")
    print(f"目标达成: {'是' if metrics['target']['target_achieved'] else '否'}")
    print("="*60)

    return metrics

def main():
    """主函数"""
    OUTPUT_DIR = os.getenv("OUTPUT_DIR", "./outputs")

    teacher_file = os.path.join(OUTPUT_DIR, "teacher_baseline_results.json")
    student_before_file = os.path.join(OUTPUT_DIR, "student_baseline_results.json")
    student_after_file = os.path.join(OUTPUT_DIR, "student_distilled_results.json")
    output_file = os.path.join(OUTPUT_DIR, "distillation_metrics.json")

    calculate_distillation_metrics(
        teacher_file,
        student_before_file,
        student_after_file,
        output_file
    )

if __name__ == "__main__":
    main()
```

## 三、批量指标计算

### 3.1 批量计算脚本

创建 `experiments/metrics/batch_calculate.py`：

```python
#!/usr/bin/env python3
"""
批量指标计算脚本
自动计算所有实验结果的指标
"""

import os
import json
from pathlib import Path
from calculate_metrics import MetricsCalculator

def find_result_files(result_dir: str):
    """查找所有结果文件"""
    result_files = []

    for file in Path(result_dir).rglob("*.json"):
        if "results" in file.name or "metrics" not in file.name:
            result_files.append(str(file))

    return result_files

def batch_calculate(result_dir: str, output_dir: str):
    """批量计算指标"""
    result_files = find_result_files(result_dir)

    print(f"找到 {len(result_files)} 个结果文件")

    os.makedirs(output_dir, exist_ok=True)

    for result_file in result_files:
        try:
            calculator = MetricsCalculator()
            calculator.load_results(result_file)

            # 生成输出文件名
            base_name = os.path.basename(result_file).replace("_results", "_metrics")
            output_file = os.path.join(output_dir, base_name)

            # 计算并保存指标
            calculator.save_metrics(output_file)
            print(f"✓ {base_name}")

        except Exception as e:
            print(f"✗ {result_file}: {e}")

def main():
    """主函数"""
    OUTPUT_DIR = os.getenv("OUTPUT_DIR", "./outputs")

    result_dir = os.path.join(OUTPUT_DIR, "results")
    output_dir = os.path.join(OUTPUT_DIR, "metrics")

    batch_calculate(result_dir, output_dir)

if __name__ == "__main__":
    main()
```

## 四、指标汇总报告

### 4.1 汇总脚本

创建 `experiments/metrics/generate_report.py`：

```python
#!/usr/bin/env python3
"""
生成指标汇总报告
"""

import os
import json
from datetime import datetime
from typing import Dict, List, Any

def collect_metrics(metrics_dir: str) -> Dict[str, Any]:
    """收集所有指标文件"""
    metrics = {}

    for file in os.listdir(metrics_dir):
        if file.endswith("_metrics.json"):
            model_name = file.replace("_metrics.json", "")

            with open(os.path.join(metrics_dir, file), 'r') as f:
                metrics[model_name] = json.load(f)

    return metrics

def generate_summary_report(metrics: Dict[str, Any], output_file: str):
    """生成汇总报告"""
    report = {
        "report_date": datetime.now().isoformat(),
        "summary": {},
        "details": metrics
    }

    # 汇总关键指标
    for model_name, model_metrics in metrics.items():
        report["summary"][model_name] = {
            "accuracy": model_metrics.get("overall", {}).get("accuracy", 0),
            "f1_macro": model_metrics.get("f1_score", {}).get("f1_macro", 0),
            "avg_inference_time": model_metrics.get("inference_time", {}).get("mean", 0)
        }

    # 保存报告
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(report, f, ensure_ascii=False, indent=2)

    # 打印汇总
    print("\n" + "="*60)
    print("指标汇总报告")
    print("="*60)

    print(f"\n{'模型':<30} {'准确率':<10} {'F1-Macro':<10} {'推理时间(s)':<12}")
    print("-"*60)

    for model_name, summary in report["summary"].items():
        print(f"{model_name:<30} {summary['accuracy']:<10.4f} {summary['f1_macro']:<10.4f} {summary['avg_inference_time']:<12.2f}")

    print("="*60)

    return report

def main():
    """主函数"""
    OUTPUT_DIR = os.getenv("OUTPUT_DIR", "./outputs")
    metrics_dir = os.path.join(OUTPUT_DIR, "metrics")
    output_file = os.path.join(OUTPUT_DIR, "metrics_summary_report.json")

    metrics = collect_metrics(metrics_dir)
    report = generate_summary_report(metrics, output_file)

    print(f"\n报告已保存到: {output_file}")

if __name__ == "__main__":
    main()
```

## 七、常见问题排查

### 问题1：指标计算异常

**现象**：指标数值不合理

**解决方案**：
1. 检查输入数据格式
2. 检查计算逻辑

1. 检查文件路径
2. 检查文件权限
3. 检查文件格式

**现象**：不同计算方式结果不同

**解决方案**：
1. 统一计算方法
