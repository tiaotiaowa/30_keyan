```python
import matplotlib.pyplot as plt
import numpy as np

def plot_main_results(metrics_summary, output_path):
    """
    绘制主结果柱状图（带误差棒）

    参数:
        metrics_summary: 指标汇总DataFrame
        output_path: 输出路径
    """
    fig, ax = plt.subplots(figsize=(10, 6))

    # 实验名称
    experiments = metrics_summary.index.tolist()
    x_pos = np.arange(len(experiments))

    # RA均值和标准差
    means = metrics_summary['RA_mean'].values
    stds = metrics_summary['RA_std'].values

    # 绘制柱状图
    bars = ax.bar(x_pos, means, yerr=stds,
                  capsize=5, alpha=0.8,
                  color=EXPERIMENT_COLORS[:len(experiments)],
                  edgecolor='black', linewidth=1.2)

    # 设置标签和标题
    ax.set_xlabel('实验配置', fontsize=12, fontweight='bold')
    ax.set_ylabel('推理准确率 (RA)', fontsize=12, fontweight='bold')
    ax.set_title('GeoKD-SR 主实验结果', fontsize=14, fontweight='bold')
    ax.set_xticks(x_pos)
    ax.set_xticklabels(experiments, rotation=45, ha='right')

    # 添加数值标签
    for i, (bar, mean, std) in enumerate(zip(bars, means, stds)):
        height = bar.get_height()
        ax.text(bar.get_x() + bar.get_width()/2., height + std,
                f'{mean:.3f}',
                ha='center', va='bottom', fontsize=9)

    # 添加网格线
    ax.yaxis.grid(True, linestyle='--', alpha=0.3)
    ax.set_axisbelow(True)

    # 设置y轴范围
    ax.set_ylim([0.5, 1.0])

    # 添加显著性标记
    add_significance_markers(ax, metrics_summary, x_pos, means, stds)

    plt.tight_layout()
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()

def add_significance_markers(ax, df, x_pos, means, stds, y_offset=0.02):
    """
    添加显著性标记

    参数:
        ax: matplotlib轴对象
        df: 数据DataFrame
        x_pos: x轴位置
        means: 均值数组
        stds: 标准差数组
        y_offset: 标记的垂直偏移
    """
    # 这里可以添加实际的显著性检验结果
    # 示例：标记与Baseline的显著差异
    baseline_mean = means[0]
    baseline_std = stds[0]

    for i in range(1, len(means)):
        if means[i] - baseline_mean > 0.05:  # 假设阈值
            # 绘制显著性标记线
            x1, x2 = x_pos[0], x_pos[i]
            y = max(means[0] + stds[0], means[i] + stds[i]) + y_offset
            h = y + 0.02

            ax.plot([x1, x1, x2, x2], [y, h, h, y], 'k-', linewidth=1)
            ax.text((x1 + x2)/2, h, '***', ha='center', va='bottom', fontsize=12)
```

```python
def plot_multi_metrics(metrics_summary, output_path):
    """
    绘制多指标对比图

    参数:
        metrics_summary: 指标汇总DataFrame
        output_path: 输出路径

    fig, axes = plt.subplots(2, 2, figsize=(12, 10))
```

### 3.1 相对提升百分比图

```python
def plot_ablation_improvement(baseline_metrics, ablation_results, output_path):
    """
    绘制消融实验相对提升图

    参数:
        baseline_metrics: 基线指标
        ablation_results: 消融实验结果
        output_path: 输出路径
    """
    fig, ax = plt.subplots(figsize=(10, 6))

    # 计算相对提升
    components = list(ablation_results.keys())
    improvements = []

    for comp in components:
        baseline_val = baseline_metrics[comp]['RA']
        ablation_val = ablation_results[comp]['RA']
        improvement = (ablation_val - baseline_val) / baseline_val * 100
        improvements.append(improvement)

    # 排序
    sorted_indices = np.argsort(improvements)
    sorted_components = [components[i] for i in sorted_indices]
    sorted_improvements = [improvements[i] for i in sorted_indices]

    # 绘制水平柱状图
    y_pos = np.arange(len(sorted_components))
    colors = [COLORS['success'] if x > 0 else COLORS['warning']
              for x in sorted_improvements]

    bars = ax.barh(y_pos, sorted_improvements, color=colors,
                   alpha=0.8, edgecolor='black', linewidth=1)

    # 设置标签
    ax.set_yticks(y_pos)
    ax.set_yticklabels(sorted_components)
    ax.set_xlabel('相对提升 (%)', fontsize=12, fontweight='bold')
    ax.set_title('消融实验：各组件贡献分析', fontsize=14, fontweight='bold')
    ax.axvline(x=0, color='black', linestyle='-', linewidth=1)
    ax.xaxis.grid(True, linestyle='--', alpha=0.3)

    # 添加数值标签
    for bar, val in zip(bars, sorted_improvements):
        width = bar.get_width()
        xpos = width + 0.5 if width > 0 else width - 0.5
        ax.text(xpos, bar.get_y() + bar.get_height()/2,
                f'{val:+.1f}%', va='center',
                ha='left' if width > 0 else 'right', fontsize=9)

    plt.tight_layout()
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()
```

### 3.2 组件重要性瀑布图

```python
def plot_component_waterfall(baseline, full_model, component_contributions, output_path):
    """
    绘制组件贡献瀑布图

    参数:
        baseline: 基线RA值
        full_model: 完整模型RA值
        component_contributions: 各组件贡献字典
        output_path: 输出路径
    """
    fig, ax = plt.subplots(figsize=(10, 6))

    # 准备数据
    components = list(component_contributions.keys())
    values = list(component_contributions.values())

    # 计算累积值
    cumulative = [baseline]
    for val in values:
        cumulative.append(cumulative[-1] + val)

    # 绘制瀑布图
    x_pos = range(len(components) + 2)
    colors = [COLORS['primary']] + \
             [COLORS['success'] if v > 0 else COLORS['warning'] for v in values] + \
             [COLORS['secondary']]

    # 绘制柱子
    for i in range(len(components) + 2):
        if i == 0:
            # Baseline
            ax.bar(i, baseline, color=colors[i], alpha=0.8,
                   edgecolor='black', linewidth=1)
        elif i == len(components) + 1:
            # Full model
            ax.bar(i, full_model, color=colors[i], alpha=0.8,
                   edgecolor='black', linewidth=1)
        else:
            # Component
            val = values[i-1]
            ax.bar(i, val, bottom=cumulative[i-1], color=colors[i],
                   alpha=0.8, edgecolor='black', linewidth=1)

    # 连接线
    ax.plot(x_pos[:-1], cumulative[:-1], 'k--', alpha=0.5)

    # 设置标签
    labels = ['Baseline'] + components + ['Full Model']
    ax.set_xticks(x_pos)
    ax.set_xticklabels(labels, rotation=45, ha='right')
    ax.set_ylabel('推理准确率 (RA)', fontsize=12, fontweight='bold')
    ax.set_title('组件贡献瀑布图', fontsize=14, fontweight='bold')
    ax.yaxis.grid(True, linestyle='--', alpha=0.3)
    ax.set_axisbelow(True)

    plt.tight_layout()
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()
```

```python
import seaborn as sns

def plot_direction_confusion_matrix(confusion_matrix, output_path):
    """
    绘制方向混淆矩阵热图

    参数:
        confusion_matrix: 8x8混淆矩阵
        output_path: 输出路径
    """
    fig, ax = plt.subplots(figsize=(8, 8))

    # 归一化
    cm_norm = confusion_matrix / confusion_matrix.sum(axis=1, keepdims=True)

    # 绘制热图
    im = ax.imshow(cm_norm, cmap='Blues', vmin=0, vmax=1)

    # 设置标签
    bearing_labels = ['N', 'NE', 'E', 'SE', 'S', 'SW', 'W', 'NW']
    ax.set_xticks(range(8))
    ax.set_yticks(range(8))
    ax.set_xticklabels(bearing_labels)
    ax.set_yticklabels(bearing_labels)

    ax.set_xlabel('预测方向', fontsize=12, fontweight='bold')
    ax.set_ylabel('真实方向', fontsize=12, fontweight='bold')
    ax.set_title('方向推理混淆矩阵', fontsize=14, fontweight='bold')

    # 添加数值标签
    for i in range(8):
        for j in range(8):
            value = cm_norm[i, j]
            text_color = 'white' if value > 0.5 else 'black'
            ax.text(j, i, f'{value:.2f}', ha='center', va='center',
                   color=text_color, fontsize=10)

    # 添加颜色条
    cbar = plt.colorbar(im, ax=ax, fraction=0.046, pad=0.04)
    cbar.set_label('准确率', fontsize=10)

    plt.tight_layout()
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()
```

```python
def plot_effect_size_forest(comparison_results, output_path):
    """
    绘制效应量森林图

    参数:
        comparison_results: 比较结果列表
        output_path: 输出路径
    """
    fig, ax = plt.subplots(figsize=(12, 8))

    # 提取数据
    comparisons = [c['comparison'] for c in comparison_results]
    effect_sizes = [c.get('cohens_d', c.get('effect_size_r', 0))
                    for c in comparison_results]
    ci_lowers = [c.get('ci_95_lower', es - 0.2) for es, c in zip(effect_sizes, comparison_results)]
    ci_uppers = [c.get('ci_95_upper', es + 0.2) for es, c in zip(effect_sizes, comparison_results)]

    # 排序
    sorted_indices = np.argsort(effect_sizes)
    y_pos = np.arange(len(comparisons))

    # 绘制森林图
    colors = [COLORS['success'] if es > 0 else COLORS['warning']
              for es in effect_sizes]

    # 误差线
    ax.errorbar(effect_sizes, y_pos,
                xerr=[np.array(effect_sizes) - np.array(ci_lowers),
                      np.array(ci_uppers) - np.array(effect_sizes)],
                fmt='o', color=colors, capsize=5, capthick=2,
                markersize=10, elinewidth=2)

    # 零线
    ax.axvline(x=0, color='black', linestyle='--', linewidth=1.5)

    # 效应量参考线
    ax.axvline(x=0.2, color='gray', linestyle=':', linewidth=1, alpha=0.5)
    ax.axvline(x=0.5, color='gray', linestyle=':', linewidth=1, alpha=0.5)
    ax.axvline(x=0.8, color='gray', linestyle=':', linewidth=1, alpha=0.5)
    ax.axvline(x=-0.2, color='gray', linestyle=':', linewidth=1, alpha=0.5)
    ax.axvline(x=-0.5, color='gray', linestyle=':', linewidth=1, alpha=0.5)
    ax.axvline(x=-0.8, color='gray', linestyle=':', linewidth=1, alpha=0.5)

    # 添加效应量标签
    ax.text(0.2, ax.get_ylim()[1] - 0.5, '小', ha='center', fontsize=8, color='gray')
    ax.text(0.5, ax.get_ylim()[1] - 0.5, '中', ha='center', fontsize=8, color='gray')
    ax.text(0.8, ax.get_ylim()[1] - 0.5, '大', ha='center', fontsize=8, color='gray')

    # 设置y轴标签
    ax.set_yticks(y_pos)
    ax.set_yticklabels(comparisons, fontsize=10)
    ax.set_xlabel("Cohen's d (效应量)", fontsize=12, fontweight='bold')
    ax.set_title('组间比较效应量森林图 (95% CI)', fontsize=14, fontweight='bold')

    # 添加网格
    ax.xaxis.grid(True, linestyle='--', alpha=0.3)
    ax.set_axisbelow(True)

    plt.tight_layout()
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()
```

```python
def plot_grouped_forest(comparison_groups, output_path):
    """
    绘制分组森林图

    参数:
        comparison_groups: 分组比较结果
            {
                '主实验': [results],
                '消融实验': [results],
                '敏感性分析': [results]
            }
        output_path: 输出路径
    """
    fig, ax = plt.subplots(figsize=(14, 10))

    y_pos = 0
    y_positions = []
    all_comparisons = []
    all_effects = []
    all_cis = []
    all_colors = []
    group_labels = []

    for group_name, results in comparison_groups.items():
        group_start = y_pos

        for result in results:
            y_positions.append(y_pos)
            all_comparisons.append(result['comparison'])
            all_effects.append(result.get('cohens_d', 0))
            all_cis.append((
                result.get('ci_95_lower', result['cohens_d'] - 0.2),
                result.get('ci_95_upper', result['cohens_d'] + 0.2)
            ))
            all_colors.append(COLORS['success'] if result.get('cohens_d', 0) > 0 else COLORS['warning'])
            group_labels.append(group_name)
            y_pos += 1

        # 添加分组间隔
        y_pos += 0.5

    # 绘制
    for i, (effect, ci, color) in enumerate(zip(all_effects, all_cis, all_colors)):
        ax.errorbar(effect, y_positions[i],
                   xerr=[[effect - ci[0]], [ci[1] - effect]],
                   fmt='o', color=color, capsize=4, markersize=8)

    # 零线
    ax.axvline(x=0, color='black', linestyle='--', linewidth=1.5)

    # 设置y轴
    ax.set_yticks(y_positions)
    ax.set_yticklabels(all_comparisons, fontsize=9)
    ax.set_xlabel("Cohen's d", fontsize=12, fontweight='bold')
    ax.set_title('分组效应量森林图', fontsize=14, fontweight='bold')

    # 添加分组标签
    ax.set_ylim(-1, y_pos)
    plt.tight_layout()
    plt.savefig(output_path, dpi=300, bbox_inches='tight')
    plt.close()
```

```python
def export_for_paper(figure_path, output_dir):
    """
    导出适合论文使用的图表

    参数:
        figure_path: 原始图表路径
        output_dir: 输出目录
    """
    from PIL import Image
    from pathlib import Path

    output_path = Path(output_dir)
    output_path.mkdir(parents=True, exist_ok=True)

    # 打开原始图像
    img = Image.open(figure_path)

    # 导出PNG（用于在线版本）
    png_path = output_path / f"{Path(figure_path).stem}_paper.png"
    img.save(png_path, dpi=300, optimize=True)

    # 导出PDF（用于印刷版本）
    pdf_path = output_path / f"{Path(figure_path).stem}_paper.pdf"
    img.convert('RGB').save(pdf_path, 'PDF', resolution=300)

    # 导出高分辨率PNG（用于演示）
    hd_path = output_path / f"{Path(figure_path).stem}_hd.png"
    img.save(hd_path, dpi=600, optimize=True)

    return png_path, pdf_path, hd_path
```

```python
def batch_export_figures(figure_dir, output_dir):
    """
    批量导出所有图表

    参数:
        figure_dir: 图表目录
        output_dir: 输出目录
    """
    from pathlib import Path

    figure_path = Path(figure_dir)
    output_path = Path(output_dir)
    output_path.mkdir(parents=True, exist_ok=True)

    for fig_file in figure_path.glob('*.png'):

    print(f"批量导出完成: {output_path}")
```
