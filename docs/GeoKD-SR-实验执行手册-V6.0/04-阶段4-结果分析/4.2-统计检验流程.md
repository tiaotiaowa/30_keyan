import pandas as pd
import numpy as np
from scipy import stats
from scipy.stats import ttest_rel, wilcoxon, shapiro, normaltest

def prepare_comparison_data(experiment_results):
    """
    准备用于比较的数据

    参数:
        experiment_results: 各实验结果字典
            {
                'exp1': {'ra': [0.72, 0.75, ...], ...},
                'exp2': {'ra': [0.78, 0.81, ...], ...},
                ...
            }

    返回:
        comparison_data: 整理后的比较数据
    """
    # 创建配对数据框
    comparison_df = pd.DataFrame()

    for exp_name, results in experiment_results.items():
        comparison_df[exp_name] = results['ra']

    return comparison_df

## 二、正态性检验

### 2.1 Shapiro-Wilk检验

**注意**: 当n=5时，Shapiro-Wilk检验的统计功效较低，需谨慎解读结果。

```python
def check_normality(data, alpha=0.05):
    """
    正态性检验

    参数:
        data: 待检验数据
        alpha: 显著性水平

    返回:
        result: 检验结果字典
    """
    # Shapiro-Wilk检验 (推荐用于小样本)
    shapiro_stat, shapiro_p = shapiro(data)

    # D'Agostino's normality test (备选)
    dagostino_stat, dagostino_p = normaltest(data)

    result = {
        'shapiro_statistic': shapiro_stat,
        'shapiro_pvalue': shapiro_p,
        'shapiro_normal': shapiro_p > alpha,
        'dagostino_statistic': dagostino_stat,
        'dagostino_pvalue': dagostino_p,
        'dagostino_normal': dagostino_p > alpha,
        'sample_size': len(data),
        'recommendation': get_normality_recommendation(data, shapiro_p, alpha)
    }

    return result

def get_normality_recommendation(data, p_value, alpha):
    """
    根据样本量和p值给出正态性判断建议
    """
    n = len(data)

    if n >= 30:
        # 大样本：中心极限定理适用
        if p_value > alpha:
            return "满足正态假设，可使用参数检验"
        else:
            return "建议使用非参数检验或考虑数据变换"
    elif n >= 15:
        # 中等样本
        if p_value > alpha:
            return "基本满足正态假设，可使用参数检验"
        else:
            return "建议优先使用非参数检验"
    else:
        # 小样本 (n < 15)
        if p_value > alpha:
            return "样本量较小，但数据未违反正态假设。谨慎使用参数检验，建议同时进行非参数检验验证"
        else:
            return "样本量小且不满足正态假设，必须使用非参数检验（Wilcoxon）"
```

### 2.2 批量正态性检验

```python
def batch_normality_test(experiment_results, metric='ra'):
    """
    对所有实验进行正态性检验

    参数:
        experiment_results: 实验结果字典
        metric: 待检验的指标名称

    返回:
        results: 检验结果汇总
    """
    results = {}

    for exp_name, data in experiment_results.items():
        metric_data = data.get(metric, [])
        if len(metric_data) > 0:
            results[exp_name] = check_normality(metric_data)

    # 创建汇总表格
    summary_data = []
    for exp_name, result in results.items():
        summary_data.append({
            '实验': exp_name,
            '样本量': result['sample_size'],
            'Shapiro统计量': f"{result['shapiro_statistic']:.4f}",
            'Shapiro p值': f"{result['shapiro_pvalue']:.4f}",
            '正态性': "是" if result['shapiro_pvalue'] > 0.05 else "否",
            '建议': result['recommendation']
        })

    summary_df = pd.DataFrame(summary_data)

    return results, summary_df
```

## 三、配对检验

### 3.1 配对t检验

适用于满足正态假设的配对数据：

```python
def paired_t_test(group1, group2, alpha=0.05):
    """
    配对t检验

    参数:
        group1: 第一组数据
        group2: 第二组数据
        alpha: 显著性水平

    返回:
        result: 检验结果
    """
    # 确保数据长度一致
    assert len(group1) == len(group2), "配对数据长度必须一致"

    # 执行配对t检验
    statistic, p_value = ttest_rel(group1, group2)

    # 计算均值差异
    mean_diff = np.mean(np.array(group1) - np.array(group2))

    # 计算差异的标准差
    std_diff = np.std(np.array(group1) - np.array(group2), ddof=1)

    # 计算效应量 Cohen's d
    cohens_d = mean_diff / std_diff if std_diff > 0 else 0

    result = {
        'test': 'Paired t-test',
        'statistic': statistic,
        'p_value': p_value,
        'significant': p_value < alpha,
        'mean_difference': mean_diff,
        'std_difference': std_diff,
        'cohens_d': cohens_d,
        'interpretation': interpret_cohens_d(cohens_d)
    }

    return result

def interpret_cohens_d(d):
    """
    解释Cohen's d效应量
    """
    abs_d = abs(d)
    if abs_d < 0.2:
        return "微小效应"
    elif abs_d < 0.5:
        return "小效应"
    elif abs_d < 0.8:
        return "中等效应"
    else:
        return "大效应"
```

### 3.2 Wilcoxon符号秩检验

适用于不满足正态假设的配对数据：

```python
def wilcoxon_test(group1, group2, alpha=0.05):
    """
    Wilcoxon符号秩检验

    参数:
        group1: 第一组数据
        group2: 第二组数据
        alpha: 显著性水平

    返回:
        result: 检验结果
    """
    # 执行Wilcoxon检验
    statistic, p_value = wilcoxon(group1, group2)

    # 计算中位数差异
    median_diff = np.median(np.array(group1) - np.array(group2))

    # 计算秩归一化效应量 (r)
    n = len(group1)
    z_score = stats.norm.ppf(1 - p_value/2)  # 近似Z分数
    r = abs(z_score) / np.sqrt(n)

    result = {
        'test': 'Wilcoxon signed-rank test',
        'statistic': statistic,
        'p_value': p_value,
        'significant': p_value < alpha,
        'median_difference': median_diff,
        'effect_size_r': r,
        'interpretation': interpret_rank_biserial(r)
    }

    return result

def interpret_rank_biserial(r):
    """
    解释秩归一化效应量
    """
    if r < 0.1:
        return "微小效应"
    elif r < 0.3:
        return "小效应"
    elif r < 0.5:
        return "中等效应"
    else:
        return "大效应"
```

### 3.3 自动选择检验方法

```python
def auto_paired_test(group1, group2, alpha=0.05):
    """
    自动选择合适的配对检验方法

    决策流程:
    1. 检查差值的正态性
    2. 根据正态性和样本量选择检验方法
    """
    differences = np.array(group1) - np.array(group2)

    # 检查差值的正态性
    _, diff_p = shapiro(differences)
    n = len(differences)

    # 决策
    if diff_p > alpha or n >= 30:
        # 使用参数检验
        result = paired_t_test(group1, group2, alpha)
        result['selection_reason'] = f"差值正态 (p={diff_p:.4f}), 使用参数检验"
    else:
        # 使用非参数检验
        result = wilcoxon_test(group1, group2, alpha)
        result['selection_reason'] = f"差值非正态 (p={diff_p:.4f}), 使用非参数检验"

    return result
```

## 四、效应量计算

### 4.1 Cohen's d 及95%置信区间

```python
def calculate_cohens_d_with_ci(group1, group2, ci=0.95):
    """
    计算Cohen's d及其置信区间

    参数:
        group1: 第一组数据
        group2: 第二组数据
        ci: 置信水平（默认0.95）

    返回:
        result: 包含d值和置信区间的结果
    """
    n1 = len(group1)
    n2 = len(group2)

    # 合并标准差
    pooled_std = np.sqrt(((n1 - 1) * np.var(group1, ddof=1) +
                          (n2 - 1) * np.var(group2, ddof=1)) /
                         (n1 + n2 - 2))

    # Cohen's d
    mean_diff = np.mean(group1) - np.mean(group2)
    d = mean_diff / pooled_std if pooled_std > 0 else 0

    # 计算置信区间
    alpha = 1 - ci
    se = np.sqrt((n1 + n2) / (n1 * n2) + d**2 / (2 * (n1 + n2)))

    # t分布临界值
    from scipy.stats import t
    t_crit = t.ppf(1 - alpha/2, n1 + n2 - 2)

    ci_lower = d - t_crit * se
    ci_upper = d + t_crit * se

    result = {
        'cohens_d': d,
        f'ci_{int(ci*100)}_lower': ci_lower,
        f'ci_{int(ci*100)}_upper': ci_upper,
        'interpretation': interpret_cohens_d(d),
        'magnitude': 'significant' if (ci_lower > 0.2 or ci_upper < -0.2) else 'negligible'
    }

    return result
```

## 五、多重比较校正

### 5.1 Holm-Bonferroni校正

适用于11组实验对比（55对比较）：

```python
def holm_bonferroni_correction(p_values, alpha=0.05):
    """
    Holm-Bonferroni多重比较校正

    参数:
        p_values: 原始p值列表
        alpha: 家族显著性水平

    返回:
        corrected: 校正后的结果
    """
    # 将p值与索引配对并排序
    indexed_p = [(i, p) for i, p in enumerate(p_values)]
    indexed_p.sort(key=lambda x: x[1])

    corrected = []
    n = len(p_values)
    rejected = []
    previous_p = 0

    for rank, (idx, p) in enumerate(indexed_p):
        # 计算校正后的alpha
        adjusted_alpha = alpha / (n - rank)

        # 判断是否拒绝原假设
        is_rejected = p < adjusted_alpha

        # 计算校正后的p值
        adjusted_p = min(p * (n - rank), 1.0)

        corrected.append({
            'comparison': idx,
            'original_p': p,
            'adjusted_p': adjusted_p,
            'adjusted_alpha': adjusted_alpha,
            'rejected': is_rejected,
            'rank': rank + 1
        })

        # 如果有一个假设未被拒绝，后续所有假设也不拒绝
        if not is_rejected:
            break

    # 恢复原始顺序
    corrected_dict = {c['comparison']: c for c in corrected}
    final_result = [corrected_dict.get(i, {'rejected': False, 'adjusted_p': 1.0})
                    for i in range(len(p_values))]

    return final_result
```

### 5.2 批量比较与校正

```python
def multiple_comparison_with_correction(experiment_results, comparisons, alpha=0.05):
    """
    执行多重比较并校正

    参数:
        experiment_results: 实验结果字典
        comparisons: 比较对列表 [('exp1', 'exp2'), ('exp1', 'exp3'), ...]
        alpha: 显著性水平

    返回:
        results: 校正后的比较结果
    """
    test_results = []
    original_p_values = []

    # 执行所有比较
    for exp1, exp2 in comparisons:
        data1 = experiment_results[exp1]['ra']
        data2 = experiment_results[exp2]['ra']

        # 自动选择检验方法
        result = auto_paired_test(data1, data2, alpha)

        test_results.append({
            'comparison': f"{exp1} vs {exp2}",
            'exp1': exp1,
            'exp2': exp2,
            **result
        })
        original_p_values.append(result['p_value'])

    # Holm-Bonferroni校正
    correction = holm_bonferroni_correction(original_p_values, alpha)

    # 合并结果
    for i, corr in enumerate(correction):
        test_results[i]['corrected_p'] = corr['adjusted_p']
        test_results[i]['corrected_significant'] = corr['rejected']

    return test_results
```

## 六、完整检验流程

### 6.1 端到端统计分析

```python
def complete_statistical_analysis(experiment_results, comparisons=None):
    """
    完整的统计分析流程

    步骤:
    1. 正态性检验
    2. 配对比较
    3. 效应量计算
    4. 多重比较校正
    5. 结果汇总

    参数:
        experiment_results: 实验结果字典
        comparisons: 比较对列表（如为None，自动生成所有配对）

    返回:
        report: 完整分析报告
    """
    report = {
        'normality': {},
        'comparisons': [],
        'summary': {}
    }

    # 1. 正态性检验
    normality_results, normality_summary = batch_normality_test(experiment_results)
    report['normality']['results'] = normality_results
    report['normality']['summary'] = normality_summary

    # 2. 生成所有比较对
    if comparisons is None:
        exp_names = list(experiment_results.keys())
        comparisons = []
        for i, exp1 in enumerate(exp_names):
            for exp2 in exp_names[i+1:]:
                comparisons.append((exp1, exp2))

    # 3. 执行比较和校正
    comparison_results = multiple_comparison_with_correction(
        experiment_results, comparisons
    )
    report['comparisons'] = comparison_results

    # 4. 汇总统计
    significant_count = sum(1 for c in comparison_results if c['corrected_significant'])
    total_count = len(comparison_results)

    report['summary'] = {
        'total_comparisons': total_count,
        'significant_comparisons': significant_count,
        'significance_rate': significant_count / total_count if total_count > 0 else 0,
        'alpha_level': 0.05,
        'correction_method': 'Holm-Bonferroni'
    }

    return report
```

### 6.2 生成分析报告

```python
def generate_statistical_report(experiment_results, output_path):
    """
    生成统计分析报告

    参数:
        experiment_results: 实验结果字典
        output_path: 报告输出路径
    """
    # 执行分析
    report = complete_statistical_analysis(experiment_results)

    # 生成Markdown报告
    md_content = "# 统计分析报告\n\n"

    # 正态性检验部分
    md_content += "## 1. 正态性检验\n\n"
    md_content += report['normality']['summary'].to_markdown(index=False)
    md_content += "\n\n"

    # 比较结果部分
    md_content += "## 2. 配对比较结果（Holm-Bonferroni校正）\n\n"
    md_content += "| 比较 | 检验方法 | 统计量 | 原始p值 | 校正p值 | 显著性 | 效应量 |\n"
    md_content += "|------|---------|--------|---------|---------|--------|--------|\n"

    for comp in report['comparisons']:
        sig_mark = "***" if comp['corrected_significant'] else "ns"
        effect_size = comp.get('cohens_d', comp.get('effect_size_r', 0))

        md_content += f"| {comp['comparison']} | {comp['test']} | "
        md_content += f"{comp['statistic']:.4f} | {comp['p_value']:.4f} | "
        md_content += f"{comp['corrected_p']:.4f} | {sig_mark} | {effect_size:.4f} |\n"

    md_content += "\n"

    # 汇总部分
    md_content += "## 3. 汇总统计\n\n"
    summary = report['summary']
    md_content += f"- 总比较数: {summary['total_comparisons']}\n"
    md_content += f"- 显著比较数: {summary['significant_comparisons']}\n"
    md_content += f"- 显著率: {summary['significance_rate']:.2%}\n"
    md_content += f"- 显著性水平: {summary['alpha_level']}\n"
    md_content += f"- 校正方法: {summary['correction_method']}\n"

    # 保存报告
    with open(output_path, 'w', encoding='utf-8') as f:
        f.write(md_content)

    return report
```

## 八、注意事项

1. **样本量问题**:
   - n=5时，统计检验功效较低，结果仅供参考
   - 建议在条件允许的情况下增加重复实验次数

2. **检验选择**:
   - 优先使用自动选择函数（auto_paired_test）
   - 对于关键结论，同时报告参数和非参数检验结果

3. **效应量报告**:
   - 总是报告效应量及其95%置信区间
   - 不要只依赖p值判断实际意义

4. **多重比较**:
   - 11组实验共55对比较，必须进行校正
   - Holm-Bonferroni比传统Bonferroni更强大
