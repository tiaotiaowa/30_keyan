### 3.4 难度评分规则 V2.0

```python
def calculate_difficulty_score_v2(spatial_type, topology_subtype=None,
                                   entity_types=None, entity_count=2):
    """
    难度评分规则 V2.0 (1.0-5.0)
    """
    # 基础分 (V2.0调整)
    base_scores = {
        "directional": 1.2,   # 原1.5
        "topological": 2.2,   # 原2.0
        "metric": 1.3,        # 原1.0
        "composite": 3.2      # 原3.0
    }

    # 拓扑子类型加成 (V2.0新增)
    topology_bonus = {
        "within": 0.0,
        "contains": 0.1,
        "adjacent": 0.3,
        "disjoint": 0.4,
        "overlap": 0.6
    }

    # 实体类型对加成 (V2.0更新)
    entity_bonus = {
        ("city", "city"): 0.0,
        ("city", "landmark"): 0.2,    # 原0.3
        ("province", "city"): 0.4,    # 原0.5
        ("river", "city"): 0.7,       # 原0.8
        ("mountain", "city"): 0.7,    # 原0.8
        ("region", "city"): 0.9       # 原1.0
    }

    # 实体数量加成 (V2.0新增)
    entity_count_bonus = max(0, (entity_count - 2) * 0.3)

    score = base_scores[spatial_type]
    if topology_subtype and topology_subtype in topology_bonus:
        score += topology_bonus[topology_subtype]
    score += entity_bonus.get(tuple(sorted(entity_types)), 0.5)
    score += entity_count_bonus

    return min(max(score, 1.0), 5.0)
```

#### 5.3.2 距离计算规则

```python
def haversine(coord1, coord2):
    """
    使用Haversine公式计算两点间的大圆距离
    """
    from math import radians, sin, cos, sqrt, atan2

    lat1, lon1 = radians(coord1[1]), radians(coord1[0])
    lat2, lon2 = radians(coord2[1]), radians(coord2[0])

    dlat = lat2 - lat1
    dlon = lon2 - lon1

    a = sin(dlat/2)**2 + cos(lat1) * cos(lat2) * sin(dlon/2)**2
    c = 2 * atan2(sqrt(a), sqrt(1-a))

    R = 6371  # 地球半径（公里）
    return R * c

def format_distance(distance_km):
    """
    格式化距离输出
    """
    if distance_km < 1:
        return f"{int(distance_km * 1000)}米"
    elif distance_km < 100:
        return f"{distance_km:.1f}公里"
    else:
        return f"约{int(round(distance_km, -1))}公里"
```

### 8.3 分布验证标准

```python
def validate_distribution(data: list) -> dict:
    """验证数据分布"""
    results = {}

    # 1. 空间关系类型分布
    rel_counts = Counter(d['spatial_relation_type'] for d in data)
    total = len(data)

    for rel_type, target_ratio in [("directional", 0.30), ("topological", 0.225),
                                    ("metric", 0.225), ("composite", 0.25)]:
        actual_ratio = rel_counts.get(rel_type, 0) / total
        deviation = abs(actual_ratio - target_ratio)
        results[f"rel_{rel_type}"] = {
            "actual": actual_ratio,
            "target": target_ratio,
            "deviation": deviation,
            "pass": deviation < 0.05  # 允许5%偏差
        }

    # 2. 难度分布
    diff_counts = Counter(d['difficulty'] for d in data)

    for diff, target_ratio in [("easy", 0.30), ("medium", 0.50), ("hard", 0.20)]:
        actual_ratio = diff_counts.get(diff, 0) / total
        deviation = abs(actual_ratio - target_ratio)
        results[f"diff_{diff}"] = {
            "actual": actual_ratio,
            "target": target_ratio,
            "deviation": deviation,
            "pass": deviation < 0.05
        }

    return results
```
